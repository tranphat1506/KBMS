\chapter{Giới thiệu và Đặt vấn đề}

\section{Giới thiệu đề tài và Đặt vấn đề}


\subsection{Lý do chọn đề tài (Motivation)}


Trong kỷ nguyên chuyển đổi số, nhu cầu về việc quản trị và khai thác tri thức đã trở nên cấp thiết hơn bao giờ hết. Các hệ quản trị cơ sở dữ liệu truyền thống (RDBMS) mặc dù rất mạnh mẽ trong việc lưu trữ và truy xuất dữ liệu có cấu trúc, nhưng vẫn còn nhiều hạn chế trong việc tự động suy diễn ra các tri thức mới từ tập dữ liệu hiện có.


Việc tích hợp trí tuệ nhân tạo, cụ thể là các hệ thống dựa trên tri thức (Knowledge-based systems), vào tầng lưu trữ dữ liệu giúp rút ngắn khoảng cách giữa "Dữ liệu thô" và "Tri thức hữu ích" [1]. Đề tài "Xây dựng Hệ quản trị Cơ sở tri thức (KBMS) dựa trên mô hình Đối tượng tính toán" được lựa chọn nhằm giải quyết bài toán này.


\subsection{Mục tiêu nghiên cứu (Research Objectives)}


Mục tiêu chính của đề tài là xây dựng một hệ thống KBMS hoàn chỉnh, có khả năng:

\begin{enumerate}
  \item \textbf{Biểu diễn tri thức chuyên sâu}: Sử dụng mô hình COKB [1] để định nghĩa các thực thể có khả năng tính toán.
  \item \textbf{Suy diễn tự động hiệu năng cao}: Áp dụng thuật toán F-Closure để tìm tập đóng tri thức một cách tối ưu.
  \item \textbf{Lưu trữ bền vững và an toàn}: Phát triển công cụ lưu trữ nhị phân hỗ trợ chỉ mục B+ Tree và nhật ký WAL [5].
  \item \textbf{Giao diện phát triển trực quan}: Cung cấp Studio IDE giúp người dùng dễ dàng thiết kế và kiểm chứng tri thức.
\end{enumerate}



\subsection{Đối tượng và Phạm vi nghiên cứu}


\begin{itemize}
  \item \textbf{Đối tượng}: Các mô hình biểu diễn tri thức, thuật toán suy diễn tiến (Forward Chaining) và các kỹ thuật quản lý CSDL.
  \item \textbf{Phạm vi}:
\begin{itemize}
  \item \textbf{Quản lý tri thức}: Thực hiện các thao tác thêm, sửa, xóa và tìm kiếm tri thức chuyên sâu qua ngôn ngữ KBQL.
  \item \textbf{Suy diễn tri thức}: Xây dựng bộ máy suy duyễn dựa trên tập luật và sự thật hiện có để sinh ra tri thức mới.
  \item \textbf{Lưu trữ}: Nghiên cứu phương pháp lưu trữ tri thức bền vững dưới cấu trúc mô hình đối tượng tính toán (COKB).
\end{itemize}
\end{itemize}



\subsection{Ý nghĩa khoa học và thực tiễn}


Đề tài góp phần chuẩn hóa phương pháp xây dựng hệ thống tri thức có khả năng mở rộng (Scale-out) và tính linh hoạt cao trong việc thay đổi luật mà không cần can thiệp vào mã nguồn ứng dụng. Đây là nền tảng quan trọng cho việc phát triển các hệ chuyên gia và hệ trợ giúp quyết định thông minh.

\chapter{Cơ sở Lý thuyết COKB}

\section{Mô hình Đối tượng Tính toán COKB}


Mô hình \textbf{COKB (Computational Objects Knowledge Base)} [1] là sự mở rộng của các hệ thống logic truyền thống, tích hợp khả năng tính toán mạnh mẽ vào các cấu trúc đối tượng, phục vụ như là định dạng lưu trữ và biểu diễn bộ não cốt lõi của hệ thống KBMS.


\subsection{Thành phần Hệ thống tri thức COKB}


Một cơ sở tri thức COKB được xác định bởi bộ 6 thành phần [1, 3]:\par
\begin{equation}
COKB = (C, H, R, Ops, Funcs, Rules)
\end{equation}


Trong đó:

\begin{itemize}
  \item \textbf{C (Concepts)}: Tập hợp các khái niệm hoặc lớp đối tượng tính toán.
  \item \textbf{H (Hierarchy)}: Các quan hệ phân cấp đặc biệt hóa giữa các khái niệm (quan hệ IS-A).
  \item \textbf{R (Relations)}: Tập các quan hệ ngữ nghĩa giữa các lớp đối tượng (ví dụ: song song, vuông góc).
  \item \textbf{Ops (Operators)}: Các toán tử tính toán trên các miền giá trị (Số thực, Vector, Ma trận).
  \item \textbf{Funcs (Functions)}: Các hàm xác định ánh xạ giữa các thuộc tính.
  \item \textbf{Rules (Rules)}: Tập hợp các luật dẫn để suy diễn ra tri thức mới.
\end{itemize}



\subsection{Mô hình Đối tượng Tính toán (Computational Object)}


Mỗi thực thể (Object) trong hệ thống được biểu diễn bởi bộ ba thành phần [2]:\par
\begin{equation}
O = (Attrs, Facts, Rules)
\end{equation}


\begin{enumerate}
  \item \textbf{Attrs (Attributes)}: Tập các thuộc tính của đối tượng. Mỗi thuộc tính bản thân nó cũng có thể là một đối tượng tính toán khác (cấu trúc đệ quy) [4].
  \item \textbf{Facts (Facts)}: Các giá trị hoặc tính chất vốn có của đối tượng đã được xác định.
  \item \textbf{Rules (Internal Rules)}: Các quy tắc, phương trình nội tại ràng buộc mối quan hệ giữa các Attrs bên trong đối tượng đó.
\end{enumerate}



\subsection{Phân cấp Khái niệm (Concept Levels)}


Trong mô hình COKB, các khái niệm được phân tầng dựa theo độ phức tạp của cấu trúc attributes:

\begin{itemize}
  \item \textbf{Cấp 0 (Base Concepts)}: Các kiểu dữ liệu cơ sở (Số thực - , Điểm - Point).
  \item \textbf{Cấp 1}: Các khái niệm xây dựng trực tiếp từ cấp 0 (Đoạn thẳng, Góc).
  \item \textbf{Cấp n}: Các khái niệm phức tạp hình thành từ các lớp thấp hơn (Tam giác, Tứ giác, Động cơ).
\end{itemize}



Việc phân cấp này giúp hệ thống quản lý tri thức theo hướng mô-đun hóa và hỗ trợ lan truyền kế thừa tri thức một cách tự động.


\section{Cơ chế Suy luận và Giải quyết vấn đề}


Dựa trên cấu trúc mô hình COKB, quá trình giải quyết bài toán thực chất là quá trình mở rộng tập sự thật dựa trên các quy tắc suy luận cưỡng bức, cho phép hệ thống "hiểu" và tự động phát sinh nội dung mà không cần người dùng nhập thủ công.


\subsection{Các Quy tắc Suy luận (Inference Rules)}


Hệ thống KBMS vận hành dựa trên 6 loại quy tắc suy luận chính (RC1 - RC6) [2]:


\begin{itemize}
  \item \textbf{RC1 (Vốn có)}: Sinh ra sự kiện từ thuộc tính mặc định của đối tượng (ví dụ: Tam giác cân có 2 góc đáy bằng nhau).
  \item \textbf{RC2 (Mặc nhiên)}: Các phép toán gán, thay thế và bắc cầu cơ bản giữa các biến và hằng số.
  \item \textbf{RC3 (Thay thế quan hệ)}: Sử dụng các quan hệ tính toán đã biết để tìm giá trị biến còn thiếu (Ohm's Law: $U, I \implies R$).
  \item \textbf{RC4 (Luật dẫn)}: Kích hoạt các luật dạng `IF <Hypothesis> THEN <Goal>`.
  \item \textbf{RC5 (Giải hệ phương trình)}: Kết hợp nhiều biến và phương trình để giải đồng thời bằng phương pháp Newton-Raphson hoặc Brent [2].
  \item \textbf{RC6 (Hành vi nội bộ)}: Suy diễn dựa trên các đặc tính cấu thành (PART-OF) [2].
\end{itemize}



\subsection{Thuật toán Tìm bao đóng (F-Closure Algorithm)}


Đây là thuật toán trọng tâm của KBMS dùng để tìm tập đóng tri thức lớn nhất có thể suy ra được từ tập giả thiết \textbf{GT}.


\textbf{Mô tả thuật thuật giải}:

\begin{enumerate}
  \item Khởi tạo tập `KnownFacts = GT`.
  \item Lặp lại cho đến khi không còn sự thật mới (Fixed-point):
\begin{itemize}
  \item Quét qua toàn bộ tập luật (Rules) và phương trình (Equations).
  \item Nếu điều kiện của luật thỏa mãn trong `KnownFacts`, đẩy kết luận vào tập đóng.
  \item Cập nhật `KnownFacts`.
\begin{enumerate}
  \item Kết quả: `FClosure(GT)` là tập hợp chứa toàn bộ tri thức tường minh và dẫn xuất.
\end{enumerate}
\end{itemize}
\end{enumerate}



\subsection{Chiến lược Giải bài toán: Heuristics}


Để tăng tốc độ suy diễn trong các không gian tri thức khổng lồ, KBMS sử dụng các quy tắc heuristics dựa trên lớp bài toán [2]:

\begin{itemize}
  \item \textbf{Heuristic 1}: Ưu tiên các luật có phần kết luận chứa biến mục tiêu (\textbf{Goal-directed}).
  \item \textbf{Heuristic 2}: Ưu tiên giải quyết các đối tượng có mức độ xác định (Confidence Level) cao nhất trước.
  \item \textbf{Heuristic 3}: Sử dụng đồ thị phụ thuộc biến để giới hạn phạm vi quét luật.
\end{itemize}

  \item \textbf{Công cụ Ontologies (Protégé):} Hỗ trợ mô hình hóa tri thức tốt nhưng hiệu năng truy vấn và khả năng tích hợp vào các hệ thống phần mềm thương mại còn hạn chế [7].

\subsection{Mục tiêu Nghiên cứu (Research Objectives)}


Dự án KBMS hướng tới việc xây dựng một hệ quản trị tri thức toàn diện với các mục tiêu cụ thể:

\begin{enumerate}
  \item \textbf{Storage Engine:} Phát triển cấu trúc cây B+ Tree nhị phân và cơ chế ghi nhật ký phục hồi (WAL) để quản lý hàng triệu thực thể tri thức bền vững.
  \item \textbf{Reasoning Engine:} Xây dựng bộ máy suy diễn tiến (Forward Chaining) dựa trên điểm đóng F-Closure cho phép tự động tính toán dữ liệu bổ sung.
  \item \textbf{Language Compiler:} Thiết kế ngôn ngữ KBQL (Knowledge Base Query Language) và bộ biên dịch (Parser/Lexer) hỗ trợ KDL (Định nghĩa) và KQL (Truy vấn).
  \item \textbf{Integrated IDE:} Phát triển môi trường Studio IDE chuyên nghiệp giúp trực quan hóa và thiết kế tri thức dễ dàng.
\end{enumerate}





\section{Tác nhân \& Vai trò Hệ thống (Actors \& Roles)}


Hệ thống KBMS được thiết kế để phục vụ các nhu cầu khác nhau từ học tập, nghiên cứu đến triển khai hệ thống công nghiệp.


\subsection{Phân loại Người dùng}


\subsubsection{a. Học viên / Nghiên cứu sinh (Researchers)}

\begin{itemize}
  \item \textbf{Nhu cầu}: Thiết kế các mô hình tri thức lý thuyết (Hình học, Y học, Hóa học).
  \item \textbf{Công cụ}: Sử dụng \textbf{KBMS Studio} làm môi trường chính để vẽ đồ thị tri thức và định nghĩa tập luật.
  \item \textbf{Hành động}: CREATE CONCEPT, INSERT FACT, SOLVE bài toán.
\end{itemize}



\subsubsection{b. Quản trị viên (System Administrators)}

\begin{itemize}
  \item \textbf{Nhu cầu}: Giám sát sức khỏe hệ thống và duy trì tính toàn vẹn của dữ liệu lớn.
  \item \textbf{Công cụ}: Sử dụng \textbf{System Dashboard} trong Studio và \textbf{CLI Management commands}.
  \item \textbf{Hành động}: REINDEX, CHECKPOINT, Quản lý Roles (GRANT/REVOKE), Giám sát RAM/Disk.
\end{itemize}



\subsubsection{c. Nhà phát triển (Developers)}

\begin{itemize}
  \item \textbf{Nhu cầu}: Tích hợp KBMS làm lõi thông minh cho các ứng dụng Expert System khác.
  \item \textbf{Công cụ}: Sử dụng \textbf{KBMS CLI} và tương tác trực tiếp qua \textbf{Binary Protocol}.
  \item \textbf{Hành động}: Bulk Insert, Tự động hóa qua Script (.kbql), Xử lý luồng kết quả Streaming.
\end{itemize}



\subsection{Mô hình Phân quyền (RBAC)}


Hệ thống sử dụng cơ chế \textbf{Role-Based Access Control} để bảo mật tri thức:

\begin{itemize}
  \item \textbf{Admin}: Toàn quyền trên mọi Cơ sở tri thức (KB) và quản lý người dùng.
  \item \textbf{Researcher}: Quyền đọc/ghi/suy diễn trên các KB được cấp phép.
  \item \textbf{Guest}: Chỉ có quyền đọc (Read-only) trên các KB công khai.

  \item \textbf{Thuật toán F-Closure}: Suy diễn tiến đến điểm đóng.
  \item \textbf{Recursive Sub-closure}: Suy diễn tổ hợp các thành phần con.

\end{itemize}
  \item \textbf{Parser \& Compiler}:
\begin{itemize}
  \item \textbf{Lexer}: Nhận diện 190+ từ khóa.
  \item \textbf{Parser}: 50+ loại AST Nodes hỗ trợ KDL, KQL, KML, TCL, KCL.
\end{itemize}
  \item \textbf{Security \& Auth}: Xác thực Base64, Role-Based Access Control (ROOT, ADMIN, RESEARCHER).

\end{itemize}

  \item \textbf{Indexing}: B+ Tree tối ưu tìm kiếm $O(\log n)$.
  \item \textbf{WAL (Write-Ahead Logging)}: Nhật ký phục hồi dữ liệu đảm bảo ACID.
  \item \textbf{Encryption}: Mã hóa tĩnh AES-256 đối với tệp `.dat` và `.kbf`.
\end{itemize}




\subsection{Bản đồ Chức năng KBQL (Detailed Registry)}


Dưới đây là danh sách đầy đủ các câu lệnh hệ thống hỗ trợ:


\subsubsection{Quản lý Kiểm soát (KCL - Security)}

\begin{itemize}
  \item `CREATE/ALTER/DROP USER`: Quản lý tài khoản người dùng hệ thống.
  \item `GRANT/REVOKE`: Phân quyền truy cập theo từng Cơ sở tri thức (KB).
\end{itemize}



\subsubsection{Định nghĩa Tri thức (KDL - Design)}

\begin{itemize}
  \item `CREATE/DROP KNOWLEDGE BASE`: Quản lý thực thể KB.
  \item `CREATE/ALTER/DROP CONCEPT`: Quản lý các lớp đối tượng và thuộc tính.
  \item `CREATE/DROP RULE \& EQUATION`: Định nghĩa luật logic và phương trình toán học.
  \item `CREATE FUNCTION/OPERATOR`: Mở rộng khả năng tính toán.
  \item `ADD/REMOVE COMPUTATION \& HIERARCHY`: Thiết lập quan hệ tính toán và kế thừa.
  \item `CREATE INDEX`: Tối ưu hóa hiệu năng truy vấn.
\end{itemize}



\subsubsection{Truy cập \& Suy diễn (KQL/RE - Solve)}

\begin{itemize}
  \item `SELECT / INSERT / UPDATE / DELETE`: Thao tác dữ liệu tri thức.
  \item `SOLVE`: Kích hoạt bộ máy suy diễn tìm lời giải cho mục tiêu.
  \item `EXPLAIN / DESCRIBE / SHOW`: Giải thích luồng suy diễn và xem cấu trúc tri thức.
\end{itemize}



\subsubsection{Bảo trì \& Hệ thống (KML - Maintenance)}

\begin{itemize}
  \item `IMPORT / EXPORT / BULK INSERT`: Di trú dữ liệu quy mô lớn.
  \item `REINDEX / CHECKPOINT / VACUUM`: Tối ưu hóa và dọn dẹp lưu trữ vật lý.
  \item `MIGRATION V2`: Nâng cấp dữ liệu từ phiên bản cũ.
\end{itemize}



\subsubsection{Quản lý Giao dịch (TCL - Transaction)}

\begin{itemize}
  \item `BEGIN / COMMIT / ROLLBACK`: Đảm bảo tính nguyên tố (Atomicity) cho các chuỗi lệnh.
\end{itemize}




\subsection{Yêu cầu Phi chức năng (Non-functional)}


\begin{itemize}
  \item \textbf{ACID Compliance}: Đảm bảo an toàn dữ liệu 100\% thông qua WAL.
  \item \textbf{High Performance}: Độ trễ < 10ms trên LAN, hỗ trợ Asynchronous I/O.
  \item \textbf{Security}: Mã hóa AES-256, Master Key Security.
  \item \textbf{Scalability}: Stateless socket handling, hỗ trợ hàng trăm kết nối đồng thời.
\end{itemize}

  \item \textbf{Chức năng chính}:

\section{Tầng Mạng (Network Layer)}


Tầng Mạng đóng vai trò là "mạch máu" truyền dẫn thông tin giữa các thành phần Application Layer và Server Layer.


\subsection{Giao thức Nhị phân (Custom Binary Protocol)}


Để tối ưu hóa hiệu năng và giảm thiểu băng thông, KBMS không sử dụng JSON/XML để giao tiếp mà sử dụng một giao thức nhị phân tùy chỉnh trên nền TCP.


\begin{itemize}
  \item \textbf{Cấu trúc Gói tin}: Gồm 1 byte `MessageType` và 4 byte `PayloadLength`, sau đó là `Payload` dữ liệu thực tế.
  \item \textbf{12+ Message Types}:
\begin{itemize}
  \item `LOGIN`: Xác thực người dùng.
  \item `QUERY`: Gửi yêu cầu KBQL.
  \item `LOGS\_STREAM`: Yêu cầu streaming log thời gian thực.
  \item `MANAGEMENT\_CMD`: Lệnh quản trị hệ thống (Dành cho ROOT).
\end{itemize}
  \item \textbf{Encoding}: Sử dụng chuẩn UTF-8 cho chuỗi văn bản và định dạng nhị phân Little-Endian cho các số nguyên/thực.
\end{itemize}



\subsection{Quản lý Kết nối (Connection Manager)}


\begin{itemize}
  \item \textbf{Asynchronous Socket I/O}: Server sử dụng mô hình lập trình bất đối xứng để phục vụ hàng trăm kết nối đồng thời với độ trễ tối thiểu.
  \item \textbf{Session Isolation}: Mỗi kết nối được gán một `SessionId` duy nhất, giúp tách biệt dữ liệu tạm thời và ngữ cảnh làm việc (Context) giữa các người dùng.
  \item \textbf{Heartbeat \& Timeout}: Đảm bảo giải phóng tài nguyên hệ thống (RAM/Socket) ngay khi kết nối bị ngắt đột ngột sau 30 giây.
\end{itemize}





\section{Tầng Máy chủ (Server Layer)}


Tầng Máy chủ (Server Layer) là thành phần phức tạp nhất của KBMS, chịu trách nhiệm cho toàn bộ quá trình biên dịch ngôn ngữ và suy diễn tự động.


\subsection{Knowledge Manager (Orchestrator)}


Knowledge Manager là bộ điều phối trung tâm của Server. Nó chịu trách nhiệm:

\begin{itemize}
  \item \textbf{KB Isolation}: Sử dụng `StoragePool` để đảm bảo mỗi người dùng có thể làm việc trên một Cơ sở tri thức độc lập mà không gây xung đột tài nguyên.
  \item \textbf{Thread-safe Operations}: Quản lý các lệnh nạp/hủy KB thông qua cơ chế khóa (Mutual Exclusion) để duy trì tính toàn vẹn của danh lục hệ thống.
  \item \textbf{Execution Routing}: Điều phối luồng làm việc giữa Parser (Biên dịch) và Inference Engine (Suy diễn).
\end{itemize}



\subsection{KBQL Parser \& Compiler}


\begin{itemize}
  \item \textbf{Phân tích Cú pháp}: Sử dụng thuật toán đệ quy xuống (Recursive Descent) để biên dịch câu lệnh KBQL thành cấu trúc AST (Abstract Syntax Tree).
  \item \textbf{Hỗ trợ}: Biên dịch các tập lệnh KDL (Định nghĩa), KQL (Truy vấn), KCL (Bảo mật), KML (Bảo trì) và TCL (Giao dịch).
\end{itemize}



\subsection{Reasoning Engine (Bộ máy Suy diễn)}


\begin{itemize}
  \item \textbf{Forward Chaining}: Thuật toán suy diễn tiến dựa trên điểm đóng \textbf{F-Closure}.
  \item \textbf{Mô hình COKB}: Thực hiện suy diễn trên các đối tượng tính toán phức tạp, giải quyết các bài toán về Quan hệ (IS-A, PART-OF) và Phương trình toán học.
\end{itemize}



\subsection{Dịch vụ Bảo mật \& Xác thực (Security \& Auth)}


Hệ thống bảo vệ tri thức thông qua cơ chế xác thực đa tầng và phân quyền dựa trên vai trò (RBAC):

\begin{itemize}
  \item \textbf{Authentication}: Đối soát thông tin đăng nhập với `UserCatalog` lưu trong System KB. Mật khẩu được mã hóa an toàn.
  \item \textbf{Authorization (RBAC)}: Phân chia 3 cấp độ truy cập:
\begin{itemize}
  \item \textbf{ROOT}: Quyền cao nhất, quản lý toàn bộ hệ thống, user và cấu hình.
  \item \textbf{ADMIN}: Quản lý các Cơ sở tri thức (KB) và Concept.
  \item \textbf{RESEARCHER}: Quyền đọc và thực thi suy diễn (Read-only / Solve).
\end{itemize}
\end{itemize}



\subsection{Hệ thống Nhật ký \& Chẩn đoán (Native Logging)}


KBMS V3 sở hữu cơ chế nhật ký độc đáo, biến log thành một dạng tri thức có thể truy vấn:

\begin{itemize}
  \item \textbf{System Logs}: Lưu trữ các sự kiện vận hành, lỗi hệ thống và hiệu năng RAM/CPU.
  \item \textbf{Audit Logs}: Ghi lại mọi câu lệnh KBQL của người dùng, thời gian thực hiện và trạng thái thành công/thất bại.
  \item \textbf{Persistence}: Mọi log đều được `SystemLogger` chèn trực tiếp vào các concept `system\_logs` và `audit\_logs` bên trong \textbf{System KB}.
  \item \textbf{Real-time Streaming}: Hỗ trợ đẩy log trực tiếp tới Studio thông qua WebSockets/TCP phục vụ giám sát thời gian thực.
\end{itemize}





\section{Tầng Lưu trữ (Storage Layer)}


Tầng Lưu trữ chịu trách nhiệm đảm bảo tính bền vững (Durability), an toàn (Security) và điều phối truy cập đồng thời cho tri thức.


\subsection{Ánh xạ Thành phần Hệ thống}


Hệ thống quản lý việc lưu trữ qua một phân cấp các Manager chuyên biệt:


\begin{figure}[H]
  \centering
  \includegraphics[width=0.75\textwidth]{assets/storage_layer_map.png}
  \caption{Bản đồ giải phẫu thành phần tầng lưu trữ vật lý.}\label{fig:4_4}
\end{figure}


\begin{itemize}
  \item \textbf{StoragePool (Singleton)}: Quản lý danh sách các KBs đang nạp và đảm bảo Thread-safety khi truy cập đa luồng.
  \item \textbf{BufferPoolManager (LRU Cache)}: Điều phối bộ nhớ đệm RAM theo từng trang (Page).
  \item \textbf{DiskManager (Physical Handler)}: Giao tiếp trực tiếp với hệ điều hành để đọc/ghi tệp nạp theo khối (16KB).
  \item \textbf{WalManager (Transaction Logger)}: Quản lý tệp nhật ký `.log` để đảm bảo ACID.
\end{itemize}



\subsection{Quản lý Trang (Physical Paging)}


\begin{itemize}
  \item \textbf{Slotted Page Layout}: Sử dụng cấu trúc trang 16KB với \textbf{24 Bytes Header} (PageId, LSN, Prev, Next, FreePtr, Count).
  \item \textbf{Hệ thống Slot}: Cho phép quản lý các bản ghi có độ dài thay đổi cực kỳ hiệu quả mà không làm phân mảnh đĩa.
\end{itemize}



\subsection{Cấu trúc Chỉ mục (B+ Tree Indexing)}


\begin{itemize}
  \item \textbf{Hiệu năng}: Tìm kiếm và truy xuất bản ghi với độ phức tạp $O(\log n)$.
  \item \textbf{Clustered Index}: Tích hợp dữ liệu trực tiếp vào các nút lá (Leaf Nodes) của cây B+ Tree để tối thiểu hóa số lần đọc đĩa.
\end{itemize}



\subsection{Điều phối Truy cập Đồng thời (Latching \& Pinning)}


Hệ thống hỗ trợ đa người dùng thao tác cùng lúc thông qua cơ chế Page Latching:

\begin{itemize}
  \item \textbf{Page Pinning}: Khi một Query cần dữ liệu, Bpm sẽ cung cấp một Frame trong RAM và tăng `PinCount`. Trang có `PinCount > 0` sẽ bị khóa vĩnh viễn trong RAM (Pinned), thuật toán LRU không thể loại bỏ nó cho tới khi được `Unpin`.
  \item \textbf{LRU Eviction}: Khi Cache đầy, các trang có `PinCount = 0` và ít được sử dụng nhất sẽ bị trục xuất (Evicted) sau khi đã được đẩy (Flush) xuống đĩa nếu là trang bẩn (Dirty).
  \item \textbf{Write-Ahead Logging (WAL)}: Mọi thay đổi dữ liệu đều được ghi nhật ký trước khi thực hiện trên RAM, đảm bảo khả năng phục hồi (Recovery) bất chấp mọi sự cố nguồn điện.
\end{itemize}



\subsection{Bảo mật Dữ liệu (Encryption)}


\begin{itemize}
  \item \textbf{AES-256}: Mã hóa toàn diện dữ liệu tĩnh (At-rest). Tệp dữ liệu thô trên đĩa hoàn toàn không thể đọc được nếu không có Master Key của hệ thống.
\end{itemize}

\subsubsection{Độ bền vững \& Nhất quán (ACID Compliance)}

\begin{itemize}
  \item \textbf{Atomicity}: Mọi thay đổi dữ liệu là nguyên tố thông qua nhật ký WAL.
  \item \textbf{Consistency}: Sau mọi lệnh KDL/KQL, tri thức phải ở trạng thái nhất quán về mặt logic.
  \item \textbf{Durability}: Dữ liệu đã cam kết (Commit) không bị mất mát sau sự cố.
\end{itemize}



\subsubsection{Hiệu năng \& Khả năng đáp ứng (Performance)}

\begin{itemize}
  \item \textbf{Độ trễ}: Thời gian phản hồi trung bình < 10ms trên mạng LAN.
  \item \textbf{Tải}: Hỗ trợ đồng thời 256+ kết nối mà không làm giảm hiệu suất suy diễn.
  \item \textbf{Truy xuất}: Tìm kiếm bản ghi $O(\log n)$ trên đĩa cứng quy mô TB.
\end{itemize}



\subsubsection{Bảo mật (Security)}

\begin{itemize}
  \item \textbf{Xác thực}: Cơ chế LOGIN nhị phân với mật khẩu băm (Hashed).
  \item \textbf{Phân quyền}: RBAC đa tầng (Role-Based Access Control) đến từng Cơ sở tri thức.
  \item \textbf{Mã hóa}: AES-256 cho dữ liệu tĩnh trên đĩa cứng.
\end{itemize}



\subsubsection{Khả năng Giám sát (Observability)}

\begin{itemize}
  \item \textbf{Logging}: Tự động ghi nhật ký mọi hành vi (Audit) và lỗi hệ thống (System Log).
  \item \textbf{Real-time}: Stream log từ Server về Studio với độ trễ < 1s.
  \item \textbf{Queryable Logs}: Cho phép dùng KBQL để truy vấn ngược lại lịch sử vận hành.
\end{itemize}

\section{Giới thiệu về Ngôn ngữ Truy vấn KBQL}


KBQL (Knowledge Base Query Language) là ngôn ngữ truy vấn chính được sử dụng trong hệ quản trị KBMS. KBQL không chỉ kế thừa các cú pháp SQL tiêu chuẩn để thao tác với dữ liệu mà còn mở rộng các khả năng suy diễn tri thức (Reasoning) dựa trên logic vị từ và tập luật (Rules).


\subsection{Triết lý Thiết kế}


KBQL được thiết kế dựa trên ba trụ cột chính:

\begin{enumerate}
  \item \textbf{Sự quen thuộc (Familiarity):} Cú pháp gần gũi với SQL giúp người dùng dễ dàng tiếp cận.
  \item \textbf{Tính tri thức (Knowledge-Driven):} Tích hợp sâu các khái niệm về Concept, Fact và Rule thay vì chỉ dừng lại ở Table/Row.
  \item \textbf{Tự động suy diễn (Automatic Inference):} Kết quả truy vấn có thể được tự động cập nhật hoặc suy luận thông qua bộ máy Suy diễn (Inference Engine) mà người dùng không cần viết logic thủ công.
\end{enumerate}

\subsection{Khái niệm Cốt lõi}


\subsubsection{Concept (Khái niệm)}

Thay vì dùng "Table", KBQL sử dụng \textbf{Concept}. Một Concept đại diện cho một thực thể hoặc một lớp đối tượng trong thế giới thực, bao gồm các biến (Variables) định nghĩa thuộc tính của nó.


\subsubsection{Fact (Sự kiện)}

Mỗi bản ghi dữ liệu trong một Concept được coi là một \textbf{Fact}. Tập hợp các Fact tạo nên cơ sở dữ liệu hiện tại.


\subsubsection{Rule (Luật)}

\textbf{Rule} định nghĩa mối quan hệ logic giữa các Fact. Khi dữ liệu (Fact) thay đổi, các Rule liên quan có thể được kích hoạt để tạo ra Fact mới hoặc cập nhật Fact hiện có (Forward Chaining).



Tiếp theo, chúng ta sẽ đi sâu vào chi tiết các lệnh định nghĩa (DDL).


\section{Định nghĩa Tri thức (KDL)}


KDL là tập hợp các lệnh dùng để định nghĩa cấu trúc, logic và các mối quan hệ trong Cơ sở tri thức (Knowledge Base).



\subsection{Quản lý Cơ sở Tri thức (Knowledge Base)}


Lệnh khởi tạo hoặc xóa bỏ một vùng chứa tri thức logic.


\begin{itemize}
  \item \textbf{CREATE KNOWLEDGE BASE <name> [DESCRIPTION "<text>"]}: Khởi tạo KB mới.
  \item \textbf{ALTER KNOWLEDGE BASE <name> SET (DESCRIPTION: "<text>")}: Cập nhật mô tả cho KB.
  \item \textbf{DROP KNOWLEDGE BASE <name>}: Xóa toàn bộ dữ liệu và cấu trúc của KB.
  \item \textbf{USE <name>}: Chuyển ngữ cảnh làm việc sang KB được chỉ định.
\end{itemize}




\subsection{Định nghĩa Khái niệm (Concept)}


Concept là thực thể cốt lõi trong KBMS, tương tự như Class trong OOP hay Table trong SQL nhưng linh hoạt hơn.


\subsubsection{Tạo Concept (CREATE CONCEPT)}

Lệnh này định nghĩa cấu trúc của một Khái niệm. Bạn có thể định nghĩa theo kiểu liệt kê biến đơn giản hoặc sử dụng các khối chuyên sâu:


\begin{verbatim}
CREATE CONCEPT <name> (
    VARIABLES (<var>: <type>, ...),
    ALIASES (<alias1>, <alias2>),
    BASE_OBJECTS (<obj1>, <obj2>),
    CONSTRAINTS (<if_then_constraints>),
    SAME_VARIABLES (<var1>, <var2>),
    CONSTRUCT_RELATIONS (<rel_definitions>),
    PROPERTIES (<prop1>, <prop2>),
    RULES (<logic_rules>),
    EQUATIONS (<math_equations>)
);
\end{verbatim}


\noindent\textit{Chi tiết các khối nâng cao:}

\begin{itemize}
  \item \textbf{ALIASES}: Định nghĩa các tên gọi khác cho Concept.
  \item \textbf{BASE\_OBJECTS}: Liệt kê các đối tượng cơ sở cấu thành nên Concept này (thường dùng trong PART\_OF).
  \item \textbf{CONSTRAINTS}: Các ràng buộc logic nội bộ phải thỏa mãn.
  \item \textbf{SAME\_VARIABLES}: Khai báo các biến ở các thành phần khác nhau nhưng trỏ cùng một giá trị tri thức.
  \item \textbf{CONSTRUCT\_RELATIONS}: Tự động tạo lập quan hệ khi dữ liệu được nạp vào.
  \item \textbf{PROPERTIES}: Các đặc tính metadata của Concept.
  \item \textbf{RULES / EQUATIONS}: Định nghĩa luật và phương trình ngay bên trong Concept thay vì tạo lệnh rời.
\end{itemize}



\subsubsection{Chỉnh sửa Concept (ALTER CONCEPT)}

Hỗ trợ thay đổi cấu trúc mà không làm mất dữ liệu hiện có.


\noindent\textit{Cấu trúc lệnh}

\begin{verbatim}
ALTER CONCEPT <name> (
    ADD (
        VARIABLE <var>: <type>,
        RULE (<rule_definition>),
        CONSTRAINT (<expression>),
        EQUATION '<expression>',
        PROPERTY <key>: <value>
    ),
    DROP (
        VARIABLE <name>,
        RULE <name>,
        CONSTRAINT <name>
    ),
    RENAME (VARIABLE <old> TO <new>)
);
\end{verbatim}




\subsection{Quản lý Phân cấp (Hierarchy)}


Định nghĩa quan hệ cha-con hoặc thành phần giữa các Concept.


\begin{itemize}
  \item \textbf{IS\_A / ISA}: Quan hệ kế thừa (ví dụ: \textit{Square IS\_A Rectangle}).
  \item \textbf{PART\_OF}: Quan hệ thành phần (ví dụ: \textit{Engine PART\_OF Car}).
\end{itemize}



\subsubsection{Cú pháp}

\begin{verbatim}
ADD HIERARCHY <child> IS_A <parent>;
ADD HIERARCHY <part> PART_OF <whole>;
REMOVE HIERARCHY <parent> IS_A <child>;
\end{verbatim}



\subsection{Định nghĩa Quan hệ (Relation)}


Mô tả các mối liên kết ngữ nghĩa giữa hai Concept.


\subsubsection{Cú pháp}

\begin{verbatim}
CREATE RELATION <name> 
FROM <domain> TO <range>
[PARAMS (<p1>, ...)]
[PROPERTIES (<transitive, symmetric, functional, ...>)];
\end{verbatim}



\subsection{Định nghĩa Luật (Rule)}


Định nghĩa logic suy diễn tự động (Forward Chaining).


\subsubsection{Cú pháp}

\begin{verbatim}
CREATE RULE <name>
SCOPE <concept>
IF <condition>
THEN <action>;
\end{verbatim}\par
\textit{Ví dụ:} `CREATE RULE FeverCheck SCOPE Patient IF temp > 37.5 THEN SET status = 'Fever';`



\subsection{Toán tử \& Hàm Tùy biến (Operator \& Function)}


Mở rộng khả năng tính toán của hệ thống.


\begin{itemize}
  \item \textbf{CREATE OPERATOR <symbol>}: Định nghĩa toán tử mới (ví dụ: `|-|`).
  \item \textbf{CREATE FUNCTION <name>}: Định nghĩa hàm con xử lý logic.
\end{itemize}



\subsubsection{Cú pháp chung}

\begin{verbatim}
CREATE {OPERATOR | FUNCTION} <identity>
PARAMS (<p1> <type1>, ...)
RETURNS <type>
BODY '<script_logic>';
\end{verbatim}



\subsection{Công thức tự động (Computation)}


Gán công thức tính toán trực tiếp vào các biến của Concept.


\subsubsection{Cú pháp}

\begin{verbatim}
ADD COMPUTATION TO <concept>
VARIABLES <inputs>, <result>
FORMULA '<expression>';
\end{verbatim}



\subsection{Chỉ mục (Index)}


Tối ưu hóa tốc độ truy vấn trên các biến cụ thể của Concept (sử dụng B+ Tree).


\subsubsection{Cú pháp}

\begin{verbatim}
CREATE INDEX <index_name> ON <concept_name> (<variable_list>);
DROP INDEX <index_name>;
\end{verbatim}



\subsection{Sự kiện tự động (Trigger)}


Tự động thực thi một câu lệnh KBQL khi có sự kiện dữ liệu xảy ra.


\subsubsection{Cú pháp}

\begin{verbatim}
CREATE TRIGGER <name>
ON ( {INSERT|UPDATE|DELETE} OF <concept> )
DO ( <kbql_statement> );
\end{verbatim}\par
\textit{Ví dụ:} `CREATE TRIGGER LogInsert ON (INSERT OF Patient) DO (PRINT 'New patient added');`


\section{Thao tác tri thức (KML)}


KML là tập hợp các lệnh dùng để chèn, cập nhật, xóa các Sự kiện (Facts) và chuyển đổi dữ liệu trong Cơ sở tri thức.



\subsection{Chèn Sự kiện (INSERT)}


Lệnh chèn một hoặc nhiều Fact vào một Concept.


\subsubsection{Chèn một dòng đơn}

\begin{verbatim}
INSERT INTO <concept_name> ATTRIBUTE (<val1>, <val2>, ...);
\end{verbatim}


\subsubsection{Chèn hàng loạt (Bulk Insert)}

Dùng để nạp dữ liệu lớn một cách nhanh chóng.\par
\begin{verbatim}
INSERT BULK INTO <concept_name> ATTRIBUTE (
    (<val1a>, <val2a>, ...),
    (<val1b>, <val2b>, ...),
    ...
);
\end{verbatim}



\subsection{Cập nhật Sự kiện (UPDATE)}


Sửa đổi giá trị các biến của các Fact hiện có dựa trên điều kiện lọc.


\subsubsection{Cú pháp}

\begin{verbatim}
UPDATE <concept_name> 
ATTRIBUTE (SET <var1>: <new_val1>, <var2>: <new_val2>) 
WHERE <conditions>;
\end{verbatim}\par
\textit{Lưu ý:} Khi cập nhật một thuộc tính, hệ thống sẽ tự động chạy lại các Luật (Rules) liên quan để đảm bảo tính nhất quán của tri thức (Consistency).



\subsection{Xóa Sự kiện (DELETE)}


Loại bỏ các Fact khỏi Concept.


\subsubsection{Cú pháp}

\begin{verbatim}
DELETE FROM <concept_name> WHERE <conditions>;
\end{verbatim}



\subsection{Chuyển đổi Dữ liệu (IMPORT / EXPORT)}


Công cụ sao lưu và phục hồi tri thức từ các tệp tin bên ngoài.


\subsubsection{Xuất dữ liệu (EXPORT)}

\begin{verbatim}
EXPORT (
    CONCEPT: <name>, 
    FILE: '<path>', 
    FORMAT: {CSV|JSON|XML}
);
\end{verbatim}


\subsubsection{Nhập dữ liệu (IMPORT)}

\begin{verbatim}
IMPORT (
    CONCEPT: <name>, 
    FILE: '<path>', 
    FORMAT: {CSV|JSON|XML}
);
\end{verbatim}



\subsection{Đặc điểm của KML trong KBMS}


Khác với các hệ quản trị CSDL thông thường, mọi thao tác KML trong KBMS đều có thể kích hoạt:

\begin{enumerate}
  \item \textbf{Reasoning Engine:} Các luật suy diễn tiến sẽ được kích hoạt ngay lập tức sau khi `INSERT` hoặc `UPDATE`.
  \item \textbf{Trigger Engine:} Các sự kiện tự động (`Trigger`) sẽ được thực thi nếu có định nghĩa cho sự kiện tương ứng.
  \item \textbf{Validation:} Dữ liệu được kiểm tra kiểu dữ liệu và các ràng buộc (Constraints) ngay tại tầng Parser và Storage.
\end{enumerate}



\section{Truy vấn và Suy diễn (KQL)}


KQL là tập hợp các lệnh dùng để lấy thông tin, thực hiện truy vấn dữ liệu và yêu cầu hệ thống thực hiện các phép suy diễn tri thức phức tạp.



\subsection{Truy vấn Dữ liệu (SELECT)}


Lệnh `SELECT` trong KBQL tương tự SQL nhưng được tối ưu để làm việc với các Concept và thuộc tính tri thức.


\subsubsection{Cú pháp đầy đủ}

\begin{verbatim}
SELECT [<columns> | * | AGGREGATE(<var>)]
FROM <concept> [AS <alias>]
[JOIN <concept> ON <condition>]
[WHERE <conditions>]
[GROUP BY <vars>] 
[HAVING <condition>]
[ORDER BY <vars> {ASC|DESC}]
[LIMIT <n> OFFSET <m>];
\end{verbatim}


\subsubsection{Các tính năng đặc biệt}

\begin{itemize}
  \item \textbf{Hàm Tổng hợp (Aggregates):} Hỗ trợ `COUNT`, `SUM`, `AVG`, `MIN`, `MAX`.
  \item \textbf{HAVING:} Mệnh đề lọc sau khi đã thực hiện `GROUP BY`.
  \item \textbf{Hàm Tính toán CALC():} Thực hiện các biểu thức toán học phức tạp ngay trong lúc truy vấn.
    \textit{Ví dụ:} `SELECT name, CALC(price * 1.1) AS price\_tax FROM Product;`
\end{itemize}




\subsection{Giải quyết Tri thức (SOLVE)}


Đây là lệnh đặc trưng của KBMS, dùng để kích hoạt bộ máy giải quyết vấn đề (Problem Solver) dựa trên các công thức và luật đã định nghĩa.


\subsubsection{Cú pháp}

\begin{verbatim}
SOLVE ON CONCEPT <name>
GIVEN <inputs>
FIND <outputs>
[SAVE]
[USING <rule_or_relation_list>];
\end{verbatim}


\subsubsection{Cách hoạt động}

\begin{enumerate}
  \item \textbf{Input:} Nạp các giá trị đã biết vào phần `GIVEN`.
  \item \textbf{Inference:} Hệ thống sử dụng Forward Chaining và Equation Solving để tìm giá trị của các biến trong phần `FIND`.
  \item \textbf{Persist (Tùy chọn):} Nếu có từ khóa `SAVE`, các kết quả vừa tìm thấy sẽ được lưu trực tiếp thành một Fact mới vào Concept tương ứng.
  \item \textbf{Result:} Trả về kết quả sau khi đã suy diễn thành công.
\end{enumerate}




\subsection{Kiểm tra Hệ thống (SHOW)}


Dùng để liệt kê các thành phần đang có trong Cơ sở Tri thức.


\begin{itemize}
  \item \textbf{SHOW CONCEPTS}: Liệt kê tất cả các Khái niệm.
  \item \textbf{SHOW RULES}: Liệt kê các luật suy diễn.
  \item \textbf{SHOW RELATIONS}: Liệt kê các quan hệ giữa các Concept.
  \item \textbf{SHOW HIERARCHIES}: Xem cấu trúc phân cấp cây tri thức.
  \item \textbf{SHOW OPERATORS / FUNCTIONS}: Xem các toán tử và hàm tùy biến.
\end{itemize}




\subsection{Phân tích \& Mô tả (DESCRIBE / EXPLAIN)}


Công cụ dành cho lập trình viên để hiểu sâu về cấu trúc và cách thực thi của hệ thống.


\begin{itemize}
  \item \textbf{DESCRIBE \{CONCEPT|KB|RULE|...\} <name>}: Hiển thị chi tiết cấu trúc định nghĩa của một đối tượng.
  \item \textbf{EXPLAIN (<kbql\_statement>)}: Hiển thị kế hoạch thực thi (Execution Plan) của một câu lệnh, rất hữu ích để tối ưu hóa truy vấn.
\end{itemize}




\subsection{Truy vấn Metadata (Hệ thống)}


Bạn có thể truy vấn trực tiếp vào danh mục hệ thống để lấy thông tin metadata.\par
\textit{Ví dụ:} `SELECT * FROM <concept\_name>.variables;` trả về danh sách các biến và kiểu dữ liệu của Concept đó.


\section{Kiểm soát truy cập (KCL)}


KCL là tập hợp các lệnh dùng để quản lý hệ thống bảo mật, tài khoản và quyền hạn truy cập của người dùng trong KBMS.



\subsection{Quản lý Người dùng (USER Management)}


KBMS cho phép định cấu hình người dùng và vai trò để bảo vệ tri thức khỏi các truy cập không mong muốn.


\subsubsection{Tạo Người dùng mới (CREATE USER)}

\begin{verbatim}
CREATE USER <username> 
PASSWORD '<password>' 
[ROLE {ADMIN|SERVICE|USER}];
\end{verbatim}


\subsubsection{Chỉnh sửa Người dùng (ALTER USER)}

Sử dụng để đổi mật khẩu hoặc trạng thái quản trị.\par
\begin{verbatim}
ALTER USER <username> (
    SET (PASSWORD: '<new_pass>', ADMIN: true)
);
\end{verbatim}


\subsubsection{Xóa Người dùng (DROP USER)}

\begin{verbatim}
DROP USER <username>;
\end{verbatim}



\subsection{Quản trị Quyền hạn (GRANT / REVOKE)}


Sử dụng để cấp hoặc thu hồi quyền thao tác trên các Concept và Tri thức cụ thể.


\subsubsection{Cấp quyền (GRANT)}

\begin{verbatim}
GRANT {SELECT, INSERT, UPDATE, DELETE, ...} 
ON CONCEPT <name> 
TO <username>;
\end{verbatim}


\subsubsection{Thu hồi quyền (REVOKE)}

\begin{verbatim}
REVOKE {SELECT, INSERT, UPDATE, DELETE, ...} 
ON CONCEPT <name> 
FROM <username>;
\end{verbatim}



\subsection{Các thực thể Quản trị đặc biệt (System Admin)}


\begin{itemize}
  \item \textbf{ADMIN}: Có toàn quyền trên tất cả các KB, Concept và người dùng khác.
  \item \textbf{SERVICE}: Có quyền truy cập vào các API hệ thống nhưng không thể thay đổi cấu trúc Metadata.
  \item \textbf{USER}: Quyền hạn mặc định, cần được cấp phép cụ thể cho từng đối tượng.
\end{itemize}



\section{TCL: Ngôn ngữ Kiểm soát Giao dịch (Transaction Control Language)}


TCL là tập hợp các lệnh dùng để quản lý sự thực thi đồng nhất của các lệnh KBQL, đảm bảo tính toàn vẹn tri thức (ACID).



\subsection{Một Giao dịch là gì? (Transaction)}


Giao dịch là một đơn vị công việc bao gồm một hoặc nhiều lệnh thực thi trên KBMS. Nếu một phần của giao dịch thất bại, toàn bộ giao dịch có thể được hủy bỏ để không làm sai lệch tri thức hiện tại.



\subsection{Các lệnh Giao dịch}


\subsection{Các lệnh Giao dịch}


\subsubsection{Bắt đầu Giao dịch (BEGIN TRANSACTION)}

\begin{verbatim}
BEGIN TRANSACTION;
\end{verbatim}\par
Từ thời điểm này, mọi thay đổi dữ liệu hoặc định nghĩa tri thức sẽ được thực thi tạm thời trong bộ đệm giao dịch.


\subsubsection{Xác nhận Giao dịch (COMMIT)}

\begin{verbatim}
COMMIT;
\end{verbatim}\par
Lệnh này xác nhận toàn bộ các thay đổi trong giao dịch và lưu trữ vĩnh viễn vào các file B+ Tree và Catalog của hệ thống.


\subsubsection{Hủy bỏ Giao dịch (ROLLBACK)}

\begin{verbatim}
ROLLBACK;
\end{verbatim}\par
Lệnh này hủy bỏ toàn bộ các lệnh từ lúc `BEGIN TRANSACTION`, đưa tri thức quay về trạng thái trước khi giao dịch bắt đầu.



\subsection{Vai trò của Giao dịch trong KBMS}


\begin{enumerate}
  \item \textbf{Tính Nguyên tử (Atomicity):} Đảm bảo một bộ các Fact liên quan được nạp vào cùng lúc (ví dụ: Thông tin bệnh nhân và các triệu chứng chẩn đoán kèm theo).
  \item \textbf{Tính Nhất quán (Consistency):} Ngăn chặn việc Rules thực hiện các thay đổi tri thức không đồng bộ giữa các Concept khác nhau.
  \item \textbf{Cách ly (Isolation):} Các giao dịch đang diễn ra không ảnh hưởng đến các truy vấn suy diễn khác cho tới khi `COMMIT` thành công.
\end{enumerate}



\section{Bảo trì Hệ thống (Maintenance)}


Hệ quản trị KBMS cung cấp một bộ các lệnh quản trị chuyên sâu giúp tối ưu hóa bộ nhớ, chỉ mục và kiểm tra tính nhất quán của tri thức.



\subsection{Lệnh Bảo trì (MAINTENANCE)}


Lệnh `MAINTENANCE` được dùng để thực hiện các tác vụ dọn dẹp và sửa lỗi hệ thống cấp thấp.


\subsubsection{Cú pháp}

\begin{verbatim}
MAINTENANCE (
    {VACUUM | REINDEX | CHECK CONSISTENCY} [ON CONCEPT <name>],
    ...
);
\end{verbatim}



\subsection{Các hành động cụ thể}


\subsubsection{Thu gọn Bộ nhớ (VACUUM)}

Khi Fact bị xóa, không gian bộ nhớ trong cây B+ Tree không được giải phóng ngay lập tức. Lệnh `VACUUM` thực hiện việc sắp xếp lại các Page dữ liệu và thu hồi dung lượng đĩa cứng dư thừa.\par
\textit{Ví dụ:} `MAINTENANCE (VACUUM);`


\subsubsection{Đánh lại Chỉ mục (REINDEX)}

Cập nhật lại toàn bộ các Index gắn với Concept. Hữu ích khi hệ thống gặp sự cố mất điện hoặc phát hiện sai lệch chỉ mục.\par
\textit{Ví dụ:} `MAINTENANCE (REINDEX ON CONCEPT Patient);`


\subsubsection{Kiểm tra Tính nhất quán (CHECK CONSISTENCY)}

Kiểm tra xem các liên kết giữa Fact, Rule và Hierarchy có còn hợp lệ hay không. Đảm bảo tri thức không bị mâu thuẫn.\par
\textit{Ví dụ:} `MAINTENANCE (CHECK CONSISTENCY ON CONCEPT (*));`



\subsection{Tần suất thực hiện bản khuyến nghị}


\begin{itemize}
  \item \textbf{VACUUM:} Sau mỗi lần xóa dữ liệu quy mô lớn (Batch delete).
  \item \textbf{REINDEX:} Định kỳ hàng tuần hoặc sau khi thực hiện việc thay đổi Metadata phức tạp.
  \item \textbf{CHECK CONSISTENCY:} Thực hiện trước khi triển khai hệ thống suy diễn vào môi trường quan trọng.
\end{itemize}



\section{Biểu thức và Hàm số (Expressions)}


Biểu thức là thành phần cốt lõi xuất hiện trong các mệnh đề `WHERE`, `IF`, `SET` và hàm `CALC()`. KBQL tích hợp một bộ máy đánh giá biểu thức mạnh mẽ hỗ trợ đầy đủ các phép toán và hàm học thuật.



\subsection{Toán tử Cơ bản (Operators)}


\subsubsection{Toán tử Số học}

\begin{itemize}
  \item `+`, `-`, `*`, `/`: Các phép toán cơ bản.
  \item `\textasciicircum{}`: Lũy thừa.
  \item `\%`: Chia lấy dư.
\end{itemize}



\subsubsection{Toán tử So sánh}

\begin{itemize}
  \item `=`, `!=` (hoặc `<>`): So sánh bằng và không bằng.
  \item `<`, `<=`, `>`, `>=`: So sánh thứ tự.
\end{itemize}



\subsubsection{Toán tử Logic}

\begin{itemize}
  \item `AND`, `OR`, `NOT`: Các phép logic Boolean.
\end{itemize}

Bên trong các biểu thức, bạn có thể tham chiếu trực tiếp đến tên các biến của Concept hiện tại. Nếu có `JOIN`, hãy sử dụng Alias để phân biệt: `e.salary * 1.1`.



\subsection{Lưu ý về Hiệu năng}


Các biểu thức phức tạp trong `WHERE` có thể làm chậm tốc độ truy vấn nếu không có Index phù hợp. Hãy sử dụng lệnh `EXPLAIN` để kiểm tra cách hệ thống đánh giá các biểu thức của bạn.


\section{Các trường hợp Chuyên gia (Expert Cases)}


Tài liệu này trình bày các tình huống sử dụng KBQL ở mức độ nâng cao, khai thác tối đa sức mạnh của bộ máy suy diễn và giải toán của KBMS.



\subsection{Giải ngược Phương trình (Inverse Problem Solving)}


Hệ thống KBMS không chỉ tính toán một chiều mà còn có khả năng giải ngược các tham số đầu vào khi biết kết quả đầu ra thông qua các thuật toán \textbf{Newton-Raphson} và \textbf{Brent}.


\subsubsection{Kịch bản: Tính toán mạch điện}

Chúng ta biết luật Ohm `U = I * R`. Nếu biết `U` và `I`, hệ thống phải tự tìm `R`.


\begin{verbatim}
CREATE CONCEPT Resistor (
    VARIABLES (u: DECIMAL, i: DECIMAL, r: DECIMAL),
    EQUATIONS (u = i * r)
);

-- Trường hợp 1: Biết I và R, tìm U (Tính xuôi)
SOLVE ON CONCEPT Resistor GIVEN i = 2, r = 10 FIND u; -- Kết quả: u = 20

-- Trường hợp 2: Biết U và I, tìm R (Tính ngược - Expert)
SOLVE ON CONCEPT Resistor GIVEN u = 220, i = 2 FIND r; -- Kết quả: r = 110
\end{verbatim}



\subsection{Suy diễn Chùm (Chained Inference)}


Thể hiện khả năng lan truyền tri thức qua nhiều lớp luật (Forward Chaining).


\subsubsection{Kịch bản: Hệ thống Cảnh báo sớm}

\begin{verbatim}
CREATE RULE R1 SCOPE Sensor IF temp > 100 THEN SET status = 'Overheat';
CREATE RULE R2 SCOPE Sensor IF status = 'Overheat' AND pressure > 50 THEN SET alert = 'Critical';
CREATE RULE R3 SCOPE Sensor IF alert = 'Critical' THEN DO (PRINT 'EMERGENCY SHUTDOWN');

-- Khi chèn dữ liệu thỏa mãn chuỗi:
INSERT INTO Sensor ATTRIBUTE (110, 60);
-- Kết quả: R1 kích hoạt -> status='Overheat' -> R2 kích hoạt -> alert='Critical' -> R3 kích hoạt -> In ra thông báo.
\end{verbatim}



\subsection{Phân cấp Kế thừa Đa tầng (Recursive Property Propagation)}


Khi định nghĩa Hierarchy, các luật ở Concept cha sẽ tự động "chảy" xuống toàn bộ các con, cháu.


\subsubsection{Kịch bản: Phân loại Hình học}

\begin{verbatim}
CREATE CONCEPT Shape (VARIABLES (color: STRING));
CREATE CONCEPT Polygon (...);
CREATE CONCEPT Triangle (...);

ADD HIERARCHY Polygon IS_A Shape;
ADD HIERARCHY Triangle IS_A Polygon;

CREATE RULE ShapeColor SCOPE Shape IF color = 'Red' THEN SET is_hot = true;

-- Khi chèn một Triangle màu đỏ:
INSERT INTO Triangle ATTRIBUTE ('Red', ...);
-- Kết quả: Mặc dù luật định nghĩa ở Shape, nhưng Triangle vẫn nhận được is_hot = true.
\end{verbatim}



\subsection{Ràng buộc nội bộ phức tạp (Internal Constraints)}


Sử dụng khối `CONSTRAINTS` để bảo vệ dữ liệu ở mức tri thức, không cho phép các Fact mâu thuẫn tồn tại.


\subsubsection{Kịch bản: Quản lý nhân sự}

\begin{verbatim}
CREATE CONCEPT Employee (
    VARIABLES (age: INT, experience: INT),
    CONSTRAINTS (
        IF age < experience + 18 THEN DO (ERROR 'Tuổi không hợp lệ so với kinh nghiệm')
    )
);

-- Lệnh này sẽ bị Parser/Engine chặn lại:
INSERT INTO Employee ATTRIBUTE (20, 5); -- (20 < 5 + 18) => Lỗi.
\end{verbatim}



\subsection{Giải hệ phương trình 2 ẩn (Newton-Raphson 2D)}


KBMS hỗ trợ giải đồng thời các hệ thức toán học.


\subsubsection{Kịch bản: Tìm giao điểm đường thẳng}

\begin{verbatim}
CREATE CONCEPT Intersection (
    VARIABLES (x: DECIMAL, y: DECIMAL),
    EQUATIONS (
        y = 2 * x + 1,
        y = -x + 4
    )
);

SOLVE ON CONCEPT Intersection FIND x, y;
-- Kết quả: x = 1, y = 3 (Hệ thống tự giải hệ phương trình).
\end{verbatim}



\subsection{Kết hợp Trigger và Rule (The Chain Reaction)}


Dùng Trigger để kích hoạt một luồng suy diễn ở một Concept hoàn toàn khác.


\subsubsection{Kịch bản: Đồng bộ kho và đơn hàng}

\begin{verbatim}
CREATE TRIGGER SyncStock ON (INSERT OF Order) 
DO (UPDATE Product ATTRIBUTE (SET stock: stock - 1) WHERE id = new.product_id);

-- Khi cập nhật kho, các luật về 'Out of stock' sẽ tự động chạy:
CREATE RULE OutOfStock SCOPE Product IF stock < 0 THEN SET status = 'Refill';
\end{verbatim}

\section{Tham chiếu Thuật ngữ Kỹ thuật (Technical Glossary)}


Để đảm bảo tính nhất quán và giúp người đọc dễ dàng tiếp cận các nội dung kỹ thuật chuyên sâu trong Chương 7, phần này định nghĩa các thuật ngữ nền tảng về hệ quản trị cơ sở dữ liệu (DBMS) được áp dụng trong KBMS V3.



\subsection{Đơn vị lưu trữ và Cấu trúc dữ liệu}


\begin{itemize}
  \item \textbf{Page (Trang):} Đơn vị lưu trữ cơ bản và nhỏ nhất mà Hệ quản trị bộ nhớ (Storage Engine) có thể đọc hoặc ghi. Trong KBMS, mỗi trang có kích thước cố định là 16KB.
  \item \textbf{Tuple (Bản ghi):} Một tập hợp dữ liệu nhị phân đại diện cho một thực thể (ví dụ: một đối tượng tam giác). Một trang có thể chứa nhiều Tuples.
  \item \textbf{Slotted Page:} Kỹ thuật tổ chức trang bằng cách sử dụng một mảng các khe cắm (Slots) ở đầu trang để trỏ đến dữ liệu nằm ở cuối trang, giúp quản lý các bản ghi có độ dài thay đổi hiệu quả.
  \item \textbf{B+ Tree:} Một cấu trúc cây tự cân bằng cho phép tìm kiếm, truy cập tuần tự, chèn và xóa dữ liệu trong thời gian logarit. Đây là cấu trúc chính để lưu trữ chỉ mục trong KBMS.
\end{itemize}




\subsection{Quản lý Bộ nhớ và I/O}


\begin{itemize}
  \item \textbf{I/O (Input/Output):} Thao tác đọc/ghi dữ liệu giữa bộ nhớ RAM và thiết bị lưu trữ vật lý (SSD/HDD). I/O là thao tác tốn kém nhất về hiệu năng.
  \item \textbf{Buffer Pool (Bộ đệm):} Một vùng nhớ RAM trung gian dùng để lưu trữ tạm thời các trang dữ liệu từ đĩa. Giúp giảm thiểu số lần phải đọc/ghi trực tiếp xuống đĩa.
  \item \textbf{LRU (Least Recently Used):} Thuật toán thay thế trang trong bộ đệm, ưu tiên giữ lại các trang được truy cập thường xuyên nhất và loại bỏ các trang ít được dùng nhất khi bộ đệm đầy.
  \item \textbf{Flush:} Hành động ghi một trang dữ liệu từ RAM xuống đĩa cứng để đảm bảo tính vĩnh cửu.
\end{itemize}




\subsection{Độ bền và An toàn dữ liệu}


\begin{itemize}
  \item \textbf{WAL (Write-Ahead Logging):} Nguyên tắc "Ghi nhật ký trước khi ghi dữ liệu". Mọi thay đổi phải được ghi vào file log trước khi cập nhật vào file dữ liệu chính để phòng ngừa mất dữ liệu khi sập nguồn.
  \item \textbf{LSN (Log Sequence Number):} Số thứ tự duy nhất được gán cho mỗi bản ghi nhật ký (Log Record) để theo dõi trình tự thời gian của các giao dịch.
  \item \textbf{ACID:} Các tính chất đảm bảo độ tin cậy của giao dịch: Nguyên tử (Atomicity), Nhất quán (Consistency), Cô lập (Isolation), và Bền vững (Durability).
  \item \textbf{AES-256:} Tiêu chuẩn mã hóa khóa đối xứng với độ dài khóa 256-bit, được KBMS dùng để bảo vệ dữ liệu tĩnh trên đĩa.
\end{itemize}




\subsection{Cơ chế chuyển đổi dữ liệu}


\begin{itemize}
  \item \textbf{Serialization (Tuần tự hóa):} Quá trình chuyển đổi cấu trúc dữ liệu hoặc trạng thái đối tượng (trong C\#) thành một định dạng (chuỗi byte) có thể lưu trữ hoặc truyền đi.
  \item \textbf{Deserialization (Giải tuần tự hóa):} Quá trình ngược lại, tái tạo đối tượng từ chuỗi byte nhị phân.
\end{itemize}

\subsubsection{Thuật toán thay thế LRU (Least Recently Used)}

KBMS sử dụng chính sách LRU để điều phối các trang trong bộ đệm:

\begin{itemize}
  \item \textbf{Pinning (Ghim trang):} Khi một trang đang được truy cập, `PinCount` tăng lên. Trang có `PinCount > 0` sẽ không bị đẩy ra khỏi RAM.
  \item \textbf{Dirty Pages:} Những trang có thay đổi dữ liệu sẽ được đánh dấu `IsDirty = true`.
  \item \textbf{Eviction (Trục xuất):} Khi bộ đệm đầy, hệ thống tìm trang có `PinCount = 0` và không được sử dụng trong thời gian lâu nhất để giải phóng. Nếu trang đó là "Dirty", nó sẽ được ghi xuống đĩa (Flush) trước khi bị thay thế.
\end{itemize}

Tiến trình xây dựng Backend cho Storage Engine được thực thi qua 4 giai đoạn:

\begin{enumerate}
  \item \textbf{Tầng Vật lý:} Xây dựng `DiskManager` hỗ trợ `Random Access` và `AES Encryption`.
  \item \textbf{Tầng Quản lý RAM:} Lập trình `BufferPoolManager` và cơ chế thay thế trang LRU.
  \item \textbf{Tầng Cấu trúc Dữ liệu:} Thiết kế giải thuật `Slotted Page` cho dữ liệu biến thiên.
  \item \textbf{Tầng Chỉ mục:} Xây dựng cấu trúc cây `B+ Tree` để kết nối hàng triệu trang thành một hệ thống đồng nhất.
\end{enumerate}



\section{Giải phẫu Trang vật lý (Page Anatomy)}


KBMS V3 sử dụng mô hình phân trang (Paging) và cấu trúc \textbf{Slotted Page} để tối ưu hóa truy xuất ngẫu nhiên và quản lý không gian trống hiệu quả cho các bản ghi có độ dài biến thiên.



\subsection{Cấu trúc nội tại của Trang (Internal Structure)}


Một trang dữ liệu chuẩn 16KB được phân rã thành 4 thành diện diện tích chính: \textbf{Header}, \textbf{Slot Array}, \textbf{Free Space}, và \textbf{Tuples}.


\begin{figure}[H]
  \centering
  \includegraphics[width=0.7\textwidth]{assets/binary_page_layout.png}
  \caption{Giải phẫu cấu trúc trang nhị phân Slotted-Page của KBMS.}\label{fig:7_2}
\end{figure}


\begin{itemize}
  \item \textbf{Header (24 Bytes)}: Nằm ở đầu trang, chứa các metadata điều phối.
  \item \textbf{Slot Array}: Danh sách các "khe cắm" trỏ tới dữ liệu, phát triển theo chiều \textbf{tiến} (từ đầu trang xuống). Mỗi slot chiếm 8 bytes: `[Offset (4B) | Length (4B)]`.
  \item \textbf{Free Space}: Vùng nhớ trống nằm giữa Slot Array và Tuples.
  \item \textbf{Tuples (Dữ liệu)}: Nội dung thực tế, được chèn theo chiều \textbf{lùi} (từ cuối trang ngược lên trên) để tận dụng tối đa không gian trống ở giữa.
\end{itemize}




\subsection{Đặc tả Header (24-Byte Layout)}


Cấu trúc Header được thiết kế để hỗ trợ tối đa cho việc phục hồi dữ liệu (Recovery) và điều hướng B+ Tree:


\begin{table}[H]
  \centering
  \begin{tabularx}{\textwidth}{|l|X|X|}
  \hline
Byte Offset & Trường dữ liệu & Ý nghĩa \\ \hline
\textbf{0 - 3} & \textbf{PageId} & ID duy nhất của trang trong tệp dữ liệu. \\ \hline
\textbf{4 - 7} & \textbf{LSN} & Log Sequence Number phục vụ cơ chế ghi nhật ký WAL. \\ \hline
\textbf{8 - 11} & \textbf{PrevPageId} & Con trỏ tới trang trước trong chuỗi lá B+ Tree. \\ \hline
\textbf{12 - 15} & \textbf{NextPageId} & Con trỏ tới trang sau phục vụ quét tuần tự (Scan). \\ \hline
\textbf{16 - 19} & \textbf{FreeSpacePointer} & Offset đánh dấu ranh giới bắt đầu của vùng dữ liệu. \\ \hline
\textbf{20 - 23} & \textbf{TupleCount} & Tổng số lượng bản ghi (Slots) hiện có trong trang. \\ \hline
  \end{tabularx}
  \caption{Bảng 11}\label{tbl:auto_tbl_11}
\end{table}



\subsection{Quản lý bản ghi biến thiên (Slotted Page Logic)}


Cơ chế này cho phép KBMS xử lý các bản ghi có kích thước khác nhau (ví dụ: một đối tượng có tên ngắn và một đối tượng có tên dài) mà không cần định nghĩa lại cấu trúc trang:


\begin{enumerate}
  \item \textbf{Chèn bản ghi:} Dữ liệu mới được đẩy vào cuối vùng `Free Space`, đồng thời một Slot mới được thêm vào đầu trang để trỏ đến vị trí đó.
  \item \textbf{Định danh RID (Record ID):} Một bản ghi được xác định duy nhất bởi cặp `(PageId, SlotId)`.
  \item \textbf{Xóa \& Thu hồi:} Khi xóa, Slot tương ứng được đánh dấu là trống. KBMS cung cấp lệnh `VACUUM` để dồn dịch dữ liệu và thu hồi không gian trống bị phân mảnh.
\end{enumerate}




\subsection{Lợi ích của kiến trúc 16KB}


\begin{enumerate}
  \item \textbf{Block Alignment:} 16KB là bội số của block size trên hầu hết các dòng SSD hiện đại, giúp tránh hiện tượng "Read-Modify-Write".
  \item \textbf{Binary Search:} Nhờ Slot Array có kích thước cố định, việc tìm kiếm một bản ghi trong trang có độ phức tạp là \textbf{O(log N)} bằng tìm kiếm nhị phân trên mảng Slot.
\end{enumerate}



\section{Chỉ mục Cây B+ Tree (Indexing Structure)}


KBMS sử dụng cây B+ Tree làm cấu trúc dữ liệu chính để quản lý và truy xuất các thực thể (`ObjectInstance`) trong mỗi Concept. B+ Tree là lựa chọn tối ưu cho việc truy vấn dữ liệu từ bộ nhớ ngoài (Disk) vì nó giảm thiểu số lần đọc đĩa (I/O) cần thiết.



\subsection{Cấu trúc Đa tầng (Tiered Structure)}


Cây B+ Tree trong KBMS V3 thường duy trì từ 3 đến 5 tầng tùy thuộc vào khối lượng tri thức:


\begin{figure}[H]
  \centering
  \includegraphics[width=0.7\textwidth]{assets/btree_graph.png}
  \caption{Sơ đồ cấu trúc đa tầng của cây B+ Tree trong việc quản lý thực thể.}\label{fig:7_3}
\end{figure}


\begin{enumerate}
  \item \textbf{Nút Gốc (Root Node):} Điểm bắt đầu của mọi truy vấn. Nếu cây chỉ có 1 trang, Root đồng thời là Leaf.
  \item \textbf{Nút Trung gian (Internal Nodes):} Chứa các khóa dẫn hướng (Keys) và con trỏ trang (`PageId`). Các nút này không chứa dữ liệu thực tế.
  \item \textbf{Nút Lá (Leaf Nodes):} Tầng cuối cùng của cây, nơi chứa toàn bộ dữ liệu của các `Tuple`. Các nút lá được liên kết với nhau bằng con trỏ `NextPageId` tạo thành một danh sách liên kết kép, cho phép quét dữ liệu theo phạm vi (Range Scan) cực nhanh.
\end{enumerate}




\subsection{Các Thuật toán Vận hành Cốt lõi}


\subsubsection{Tìm kiếm (Search) - $O(\log_b n)$}

Hệ thống bắt đầu từ Root, so sánh khóa cần tìm với các ranh giới trong Nút trung gian để chọn `PageId` tiếp theo, cho đến khi chạm tới Nút lá. Tại đây, `Slotted Page` sẽ thực hiện tìm kiếm nhị phân trên `Slot Array` để lấy dữ liệu.


\subsubsection{Chèn và Tách trang (Insert \& Split)}

Khi một trang bị đầy (Overflow):

\begin{enumerate}
  \item \textbf{Tách trang:} Một trang mới được cấp phát.
  \item \textbf{Phân bổ:} Một nửa số lượng bản ghi được chuyển sang trang mới.
  \item \textbf{Cập nhật cha:} Khóa phân tách được chèn vào Nút trung gian cấp trên. Nếu Nút cha cũng đầy, quá trình tách sẽ đệ quy ngược lên trên, có thể làm tăng chiều cao của cây.
\end{enumerate}



\subsubsection{Xóa và Gộp trang (Delete \& Merge)}

Khi số lượng bản ghi trong một trang giảm xuống dưới 50\% (Underflow):

\begin{itemize}
  \item \textbf{Vay mượn (Borrow):} Hệ thống kiểm tra các trang lân cận để chuyển bớt bản ghi sang.
  \item \textbf{Gộp trang (Merge):} Nếu không thể vay mượn, hai trang lân cận sẽ được gộp làm một và trang thừa sẽ được giải phóng về cho `DiskManager`.
\end{itemize}




\subsection{Clustered Index trong KBMS}


Khác với các hệ quản trị CSDL truyền thống thường tách rời Index và Data, KBMS V3 sử dụng mô hình \textbf{Clustered Index}:

\begin{itemize}
  \item Dữ liệu thực tế của `ObjectInstance` nằm trực tiếp tại các Nút lá của cây.
  \item \textbf{Lợi ích:} Tiết kiệm một lần truy xuất đĩa (Disk Seek) vì sau khi tìm thấy khóa trong Index, hệ thống có ngay dữ liệu mà không cần trỏ đến một file dữ liệu riêng biệt.
  \item \textbf{Hiệu năng:} Cực kỳ hiệu quả cho các truy vấn theo dải (ví dụ: `SELECT * FROM TamGiac WHERE canh\_a > 5`).
\end{itemize}

\section{Ánh xạ Mô hình sang Dữ liệu nhị phân (Model Serialization)}


Cỗ máy lưu trữ V3 (Storage Engine V3) đóng vai trò là xương sống bền vững cho KBMS, thực hiện việc chuyển đổi hai chiều giữa các thực thể lập trình (C\# Objects) và các khối nhị phân lưu trữ trên đĩa (Binary Pages).


\subsection{Chiến lược Nhất quán và Linh hoạt}


Hệ thống phân tách chiến lược tuần tự hóa thành hai luồng chính dựa trên đặc thù dữ liệu:


\begin{enumerate}
  \item \textbf{Dữ liệu mật độ cao (ObjectInstance):} Sử dụng định dạng `Tuple` nhị phân nén với bảng Offset để tối ưu hóa hiệu năng quét (Scan) và tìm kiếm (Index).
  \item \textbf{Siêu dữ liệu \& Lược đồ (Concept, Relation, Rule, KB):} Sử dụng định dạng JSON-to-Binary (UTF-8) để duy trì tính linh hoạt tối đa khi cấu trúc mô hình tri thức thay đổi.
\end{enumerate}




\subsection{Đặc tả Kỹ thuật của `ObjectInstance` Tuple}


Mỗi bản ghi đối tượng (`ObjectInstance`) được gói gọn trong một đơn vị `Tuple`. Cấu trúc nhị phân của nó tuân thủ sơ đồ phân bổ bộ nhớ sau:


\begin{table}[H]
  \centering

  \hline
Thành phần & Kích thước & Mô tả \\ \hline
\textbf{Field Count} & 2 Bytes & Số lượng trường dữ liệu có trong bản ghi. \\ \hline
\textbf{Offsets Array} & $N \times 2$ Bytes & Mảng vị trí kết thúc của từng trường (tính từ đầu Tuple). \\ \hline
\textbf{Data Payload} & Biến thiên & Chứa dữ liệu thực tế của các trường đã được mã hóa. \\ \hline
  \end{tabularx}
  \caption{Bảng 12}\label{tbl:auto_tbl_12}
\end{table}


\subsubsection{Ánh xạ Logic trường dữ liệu}

Trong KBMS V3, các trường dữ liệu của một `ObjectInstance` được sắp xếp theo thứ tự cố định:

\begin{itemize}
  \item \textbf{Trường 0 (ID):} Chứa `GUID` (16 bytes) định danh duy nhất cho đối tượng.
  \item \textbf{Trường 1 (Schema):} Một chuỗi ký tự UTF-8 chứa tên các biến (Variables) ngăn cách bởi dấu gạch đứng `|`. Đây là "snapshot" của lược đồ tại thời điểm ghi.
  \item \textbf{Trường 2..N:} Chứa giá trị của các biến tương ứng dưới dạng chuỗi UTF-8.
\end{itemize}





\subsection{Tuần tự hóa Siêu dữ liệu (Metadata Serialization)}


Các đối tượng mang tính định nghĩa (Definitions) được quản lý bởi các Catalog chuyên biệt thông qua gói `System.Text.Json`:


\subsubsection{Quy trình lưu trữ Concept \& KnowledgeBase}

\begin{enumerate}
  \item \textbf{Chuyển đổi:} Đối tượng C\# $\rightarrow$ Chuỗi JSON $\rightarrow$ Mảng Byte (UTF-8).
  \item \textbf{Đóng gói:} Mảng byte này được coi như một `Tuple` duy nhất và chèn vào các trang thuộc danh mục hệ thống (`catalog:concepts` hoặc `catalog:kbs`).
  \item \textbf{Phân vùng:}
\begin{itemize}
  \item `KbCatalog`: Quản lý danh sách các cơ sở tri thức toàn cục.
  \item `ConceptCatalog`: Quản lý lược đồ tri thức riêng cho từng KB.
\end{itemize}
\end{enumerate}




\subsection{Cơ chế Phục hồi Kiểu dữ liệu (Hydration \& Casting)}


Vì tầng lưu trữ nhị phân xử lý dữ liệu thô (Raw bytes/Strings) để đảm bảo tốc độ, việc khôi phục các kiểu dữ liệu logic (Strong Typing) được thực hiện ở tầng `V3DataRouter`:


\begin{enumerate}
  \item \textbf{Giải nén:} Đọc `Tuple` từ trang `SlottedPage`, tách các trường dựa trên bảng Offset.
  \item \textbf{Khớp dữ liệu:} Lấy danh sách tên biến từ Trường 1 (Schema).
  \item \textbf{Ép kiểu (Late-Binding Casting):} Đối chiếu với định nghĩa của `Concept` hiện tại để chuyển đổi chuỗi sang kiểu dữ liệu đích:
\begin{itemize}
  \item `INT`/`LONG` $\rightarrow$ `long.Parse()`
  \item `FLOAT`/`DOUBLE` $\rightarrow$ `double.Parse()`
  \item `BOOLEAN` $\rightarrow$ `bool.Parse()`
  \item `DECIMAL` $\rightarrow$ `decimal.Parse()`
\begin{enumerate}
  \item \textbf{Xử lý tiến hóa:} Nếu Concept có thêm biến mới chưa có trong dữ liệu cũ, giá trị `null` sẽ được gán tự động để bảo toàn logic suy diễn.
\end{enumerate}
\end{itemize}
\end{enumerate}

\section{Độ bền \& Nhật ký ghi trước (Durability \& WAL)}


Đảm bảo dữ liệu không bị mất mát khi hệ thống bị ngắt điện đột ngột là nhiệm vụ của `WalManager`. Thành phần này thực hiện các giao thức để đảm bảo tính \textbf{Durability} (Độ bền vững) trong mô hình ACID.



\subsection{Giao thức Write-Ahead Logging (WAL)}


KBMS tuân thủ nguyên tắc vàng của hệ quản trị cơ sở dữ liệu: \textbf{Ghi Nhật ký trước khi ghi Dữ liệu}.


\begin{enumerate}
  \item \textbf{Log First:} Mọi thay đổi (Insert, Update, Delete) được mô tả thành một đối tượng `LogRecord` và ghi vào tệp `.log` trên đĩa trước khi dữ liệu thực tế được cập nhật vào Buffer Pool.
  \item \textbf{Sequential I/O:} Việc ghi Log là thao tác ghi tuần tự (Sequential), giúp tận dụng tối đa băng thông của ổ cứng mà không bị trễ do tìm kiếm đầu đọc (Disk Seek).
  \item \textbf{Atomic Write:} Một giao dịch chỉ được coi là thành công khi `FileStream.Flush(true)` xác nhận dữ liệu Log đã nằm an toàn trên bề mặt vật lý của đĩa.
\end{enumerate}




\subsection{Số tuần tự nhật ký (LSN - Log Sequence Number)}


LSN là "dấu vân tay thời gian" của mọi thay đổi trong hệ thống:

\begin{itemize}
  \item Mỗi bản ghi Log có một LSN tăng dần duy nhất.
  \item Mỗi trang dữ liệu (`Page`) lưu trữ LSN của thao tác ghi cuối cùng tác động lên nó (tại Offset 4-7 trong Header).
  \item \textbf{Quy tắc an toàn:} Một trang dữ liệu chỉ được phép ghi xuống đĩa (Flush) nếu LSN của trang đó nhỏ hơn hoặc bằng LSN lớn nhất đã được ghi thành công vào file Log.
\end{itemize}




\subsection{Quy trình Phục hồi sau sự cố (Recovery)}


Khi KBMS khởi động lại sau một vụ sập nguồn đột ngột (Crash), nó thực hiện quy trình \textbf{Forward Recovery}:


\begin{figure}[H]
  \centering
  \includegraphics[width=0.7\textwidth]{assets/recovery_flow.png}
  \caption{Cơ chế tái hiện (Redo) dữ liệu từ nhật ký WAL để khôi phục trạng thái nhất quán.}\label{fig:7_5}
\end{figure}


\begin{enumerate}
  \item \textbf{Phân tích (Analysis):} Xác định vị trí bản ghi Log cuối cùng và điểm kiểm tra (Checkpoint) gần nhất.
  \item \textbf{Tái hiện (Redo Phase):} Hệ thống đọc file Log và thực hiện lại các thao tác nếu `LSN\_Log > LSN\_Page`. Điều này đảm bảo mọi giao dịch đã được xác nhận (Committed) nhưng chưa kịp ghi xuống file `.dat` sẽ được phục hồi đầy đủ.
  \item \textbf{Nhất quán:} Sau khi hoàn tất, hệ thống sẵn sàng phục vụ các truy vấn mới.
\end{enumerate}




\subsection{Cơ chế Checkpoint}


Để ngăn file Log phình to vô hạn, định kỳ KBMS thực hiện \textbf{Checkpoint}:

\begin{itemize}
  \item Ghi toàn bộ các trang "Dirty" từ Buffer Pool xuống đĩa.
  \item Đánh dấu vị trí an toàn trong file Log.
  \item Xóa bỏ các bản ghi Log cũ đã được lưu trữ an toàn trên file dữ liệu chính, giúp giải phóng không gian lưu trữ.
\end{itemize}



\section{Quản lý Vật lý \& Mã hóa (Physical Disk Manager)}


Tầng thấp nhất của Storage Engine chịu trách nhiệm giao tiếp trực tiếp với hệ điều hành và bảo vệ dữ liệu tĩnh (Data-at-Rest) thông qua các thuật toán mã hóa tiên tiến.



\subsection{Ánh xạ Trang vật lý (Physical Block Mapping)}


KBMS V3 không lưu trữ dữ liệu dưới dạng luồng liên tục mà chia thành các khối cố định (Blocks) trên đĩa.


\begin{itemize}
  \item \textbf{Page ID to Offset:} Mọi yêu cầu truy xuất trang `PageId` được `DiskManager` chuyển đổi thành vị trí chính xác trong file thông qua công thức:
    \begin{equation}
Offset = PageId \times 16416
\end{equation}
  \item \textbf{Truy cập ngẫu nhiên ($O(1)$):} Sử dụng phương thức `FileStream.Seek` để nhảy trực tiếp đến khối dữ liệu cần thiết mà không phải quét toàn bộ tệp, đảm bảo hiệu năng ổn định ngay cả khi tệp cơ sở tri thức lên đến hàng trăm Gigabytes.
\end{itemize}




\subsection{Mã hóa dữ liệu tĩnh (AES-256 Encryption)}


Mọi dữ liệu trong KBMS V3 đều được mã hóa ở mức trang (Page-level encryption) trước khi được ghi xuống đĩa cứng để ngăn chặn việc đánh cắp dữ liệu vật lý.


\begin{itemize}
  \item \textbf{Thuật toán:} AES-256 (Advanced Encryption Standard).
  \item \textbf{Quy trình ghi:}
\begin{enumerate}
  \item Dữ liệu 16KB được lấy từ Buffer Pool.
  \item Mã hóa bằng `MasterKey` của KB.
  \item Thêm 32 bytes IV (Initialization Vector) và Padding.
  \item Ghi khối \textbf{16,416 bytes} xuống đĩa.
\begin{itemize}
  \item \textbf{Quy trình đọc:} Thực hiện ngược lại để giải mã và đưa dữ liệu thô vào RAM.
\end{itemize}
\end{enumerate}
\end{itemize}

\subsection{Nguyên lý Đóng kín Tri thức (Knowledge Closure)}


Trực giác của KBMS dựa trên khái niệm \textbf{Tập đóng (Closure)}. Cho một tập hợp các sự thật đầu vào (\textbf{GT}) và một tập hợp các luật/phương trình (\textbf{Rules}), mục tiêu là tìm ra tập hợp lớn nhất các sự thật có thể được suy ra một cách logic.


\begin{equation}
Closure(GT, Rules) = GT \cup \{ \text{Kết luận từ các Luật và Phương trình được kích hoạt} \}
\end{equation}


Hệ thống sẽ lặp lại quá trình suy diễn cho đến khi đạt được trạng thái \textbf{Fixed-point} (điểm dừng) - nơi việc tiếp tục suy diễn không sinh ra thêm bất kỳ sự thật mới nào [6]. Việc tìm tập đóng này dựa trên thuật toán F-Closure tối ưu cho các vật thể tính toán [1].



\subsection{Biểu diễn Tri thức (Knowledge Representation)}


Trong KBMS, tri thức được biểu diễn dưới dạng các đối tượng (Concepts) và các ràng buộc (Constraints) giữa chúng.


\subsubsection{Các thành phần của một Concept}

\begin{itemize}
  \item \textbf{Variables (Biến):} Các thuộc tính của đối tượng (ví dụ: `Price`, `Quantity`).
  \item \textbf{Rules (Luật):} Logic `IF-THEN` để suy diễn ra giá trị hoặc sự kiện mới.
  \item \textbf{Equations (Phương trình):} Các công thức toán học mô tả mối quan hệ giữa các biến.
  \item \textbf{Constraints (Ràng buộc):} Các điều kiện mà dữ liệu buộc phải thỏa mãn.
  \item \textbf{Relations (Quan hệ):} Mối liên kết giữa các Concept (IS-A, PART-OF, ...).
\end{itemize}




\subsection{Quy trình Suy diễn (Reasoning Lifecycle)}


Khi một bản ghi dữ liệu (Fact) được đưa vào hệ thống, bộ máy suy diễn thực hiện các bước sau:


\begin{enumerate}
  \item \textbf{Preprocessing (Tiền xử lý):} Nạp toàn bộ metadata của Concept, bao gồm các luật kế thừa từ IS-A và Part-Of.
  \item \textbf{Initialization (Khởi tạo):} Đưa các Fact đầu vào vào tập \textbf{GT (Ground Truth)}.
  \item \textbf{Cyclic Evaluation (Đánh giá vòng lặp):} Hệ thống liên tục quét qua tập hợp các Rules và Equations cho đến khi không còn Fact nào mới được sinh ra.
  \item \textbf{Verification (Kiểm chứng):} Sau khi suy diễn xong, hệ thống kiểm tra lại toàn bộ các `Constraints` để đảm bảo kết quả suy diễn không vi phạm bất kỳ quy tắc nào.
\end{enumerate}




\subsection{Chiến lược Suy diễn: Forward vs. Backward Chaining}


Trong lý thuyết Hệ chuyên gia, có hai chiến lược chính:

\begin{itemize}
  \item \textbf{Forward Chaining (Suy diễn tiến)}: Từ sự thật (\textbf{GT}) suy ra kết luận. Thích hợp cho việc "khám phá" mọi tri thức có thể từ một tập dữ liệu nhỏ.
  \item \textbf{Backward Chaining (Suy diễn lùi)}: Từ mục tiêu (\textbf{KL}) tìm ngược lại xem cần những sự thật nào. Thích hợp cho việc "kiểm chứng" giả thuyết trong một cơ sở tri thức khổng lồ.
\end{itemize}



\subsubsection{Tại sao KBMS V3 chọn Forward Chaining?}

Hệ thống KBMS hiện đại được tối ưu hóa cho các \textbf{Mô hình Vật thể Tính toán (Computational Objects)}. Trong các mô hình này (như Hình học hay Tài chính), các biến số và công thức có mối quan hệ ràng buộc chặt chẽ và cục bộ. Việc sử dụng \textbf{Goal-Directed Forward Chaining} mang lại các ưu điểm:

\begin{enumerate}
  \item \textbf{Hiệu năng}: Tìm tập đóng (Closure) nhanh hơn việc duyệt cây mục tiêu đệ quy (SLD-resolution) trong các hệ thống chứa nhiều phương trình toán học.
  \item \textbf{Tính đầy đủ}: Đảm bảo mọi hệ quả của một thay đổi dữ liệu đều được cập nhật vào GT.
  \item \textbf{Dừng sớm (Early Exit)}: Mặc dù là suy diễn tiến, Server sẽ dừng ngay lập tức khi tất cả các biến mục tiêu trong danh sách \textbf{KL} đã được tìm thấy (Dòng code 378).
\end{enumerate}



\subsubsection{Giới hạn An toàn (Safety Limit)}

Để ngăn chặn các vòng lặp vô hạn trong trường hợp tri thức bị mâu thuẫn hoặc đệ quy quá sâu, bộ máy suy diễn áp dụng giới hạn: \textbf{Tối đa 50 vòng lặp (Iterations)}. Nếu sau 50 lần quét mà hệ thống vẫn sinh ra Fact mới, nó sẽ tự động dừng và báo cáo trạng thái Closure hiện tại (Dòng code 93).



\subsubsection{IS-A (Kế thừa)}

\begin{itemize}
  \item \textbf{Ý tưởng:} Nếu A \textbf{IS-A} B, thì mọi biến và luật của B sẽ được "truyền" sang A.
  \item \textbf{Ví dụ:} `Square` IS-A `Rectangle`. Khi tính diện tích cho `Square`, hệ thống có thể dùng luôn luật `Area = Width * Height` của `Rectangle`.
\end{itemize}



\subsubsection{PART-OF (Thành phần)}

\begin{itemize}
  \item \textbf{Ý tưởng:} Mô tả cấu trúc phân rã của đối tượng. Một đối tượng lớn được tạo từ nhiều đối tượng con.
  \item \textbf{Ví dụ:} `Car` PART-OF `Engine`, `Wheel`. Hệ thống tự động ánh xạ các biến từ các phần con vào đối tượng cha để suy diễn tổng thể.
\end{itemize}



\subsubsection{CONSTRUCT\_RELATIONS (Quy tắc quan hệ)}

\begin{itemize}
  \item \textbf{Ý tưởng:} Định nghĩa các quy luật chung cho một mối quan hệ cho trước (ví dụ: Quan hệ `CongRuence` - Bằng nhau trong hình học).
  \item \textbf{Ứng dụng:} Khi hai đối tượng thỏa mãn quan hệ này, các phương trình tương ứng (như `Object1.Size = Object2.Size`) sẽ được tự động tiêm vào hệ thống để giải.
\end{itemize}




\subsection{Giải trình và Truy vết (Traceability)}


Bộ máy suy diễn KBMS không phải là một "Black Box" (Hộp đen). Nó cung cấp khả năng giải thích lý do tại sao một giá trị được tạo ra.

\begin{itemize}
  \item \textbf{Derivation Trace:} Lưu trữ nguồn gốc (Source) của mỗi biến được suy diễn, bao gồm cả các biến đầu vào được dùng tại bước đó.
  \item \textbf{Cơ chế:} Mỗi lần một Rule hoặc Equation kích hoạt thành công, một bản ghi Trace sẽ được đẩy vào kết quả truy vấn.
\end{itemize}




\subsection{Kế hoạch Triển khai Mã nguồn Engine (Implementation Strategy)}


Để hiện thực hóa bản thiết kế kỹ thuật trên thành phân hệ phần mềm hoạt động thực tế, lộ trình triển khai (Backend C\#) được chia làm 4 giai đoạn chính (Phases):


\subsubsection{Giai đoạn 1: Tiền xử lý \& Xây dựng Cấu trúc Tri thức (Preprocessing)}

Xây dựng lớp Abstract Syntax Tree (AST) tiếp nhận từ bộ phân tích (Parser). Ở giai đoạn này, dữ liệu đầu vào (Sự thật - Fact) được "Làm phẳng" (Flattening) bằng cách quét chuỗi kế thừa `IS-A` và `PART-OF` nhằm lôi kéo tất cả Metadata cần thiết sang đối tượng con. Quá trình tiền xử lý biến danh mục tri thức thô thành một đối tượng sẵn sàng chạy (Executable Concept).


\subsubsection{Giai đoạn 2: Lõi Suy diễn Lan truyền (Deductive Core \& Propagation)}

Xây dựng vòng lặp chính của cơ chế \textbf{Goal-Directed Forward Chaining} (FClosure).

\begin{itemize}
  \item Cài đặt luồng nhận diện chuỗi biến tương đương và sao chép/lan truyền giá trị tức thời (\textbf{SameVariables Propagation}).
  \item Chạy luồng khớp mệnh đề IF-THEN (\textbf{Rule Matching}) để đẩy liên tiếp kết luận logic vào Tập Sự Thật (Ground Truth).
\end{itemize}



\subsubsection{Giai đoạn 3: Tích hợp Lõi giải Toán học (Mathematical Solvers)}

Phát triển các Adapter kết nối lớp thuật toán tính toán với hệ thống:

\begin{itemize}
  \item Adapter Phương trình 1 ẩn (\textbf{Brent's Method}): Quét qua các biểu thức còn 1 biến và tiến hành tính toán hội tụ nghiệm số tự động.
  \item Adapter Hệ phương trình (\textbf{Newton-Raphson 2D}): Nhận diện cụm 2 phương trình chung 2 ẩn, định cấu hình ma trận Jacobian và cập nhật vòng lặp đệ quy.
\end{itemize}



\subsubsection{Giai đoạn 4: Bảo vệ Hệ thống \& Truy vết (Safety Constraints \& Traceability)}

Đảm bảo Engine chạy độc lập không gây vắt kiệt RAM máy chủ bằng cách cài giới hạn an toàn \textbf{$N=50$ vòng lặp tối đa (Iterations)}. Nếu sau 50 chu kỳ duyệt hệ thống vẫn chưa trả về điểm đóng (Closure), thuật toán tự ngắt kết nối. Song song đó, viết bộ thu thập nhật ký truy xuất (\textbf{Trace System}) để xuất định dạng dữ liệu cây giải trình thành dạng JSON chuyển qua cho Front-End UI hiển thị.



\subsection{Các Thuật toán Suy diễn Cốt lõi (Forward Chaining - FClosure)}


Thuật toán này là xương sống của KBMS, thực hiện việc mở rộng tập tri thức GT cho đến khi đạt điểm đóng (Closure).


\subsubsection{Các Giai đoạn Thực thi (Phases)}

Bên trong vòng lặp chính, bộ máy thực hiện các bước sau theo thứ tự ưu tiên:


\begin{enumerate}
  \item \textbf{Recursive Sub-closure (Phase 6.2)}: Duyệt qua các biến có kiểu dữ liệu là một Concept khác. Hệ thống sẽ tự động gọi hồi quy `FindClosure` cho các đối tượng con này để đồng bộ tri thức từ dưới lên.
  \item \textbf{SameVariables Propagation (RC2)}: Lan truyền giá trị giữa các cặp biến được đánh dấu tương đương. Ví dụ: `a = b` thì nếu biết `a`, hệ thống tự động gán giá trị cho `b`.
  \item \textbf{Computation Relations (RC3)}: Thực thi các quan hệ tính toán trực tiếp (\textbf{Rf}) từ biểu thức.
  \item \textbf{Logic Rules (RC4)}: Khớp mẫu các luật `IF-THEN`. Nếu mẫu khớp, các kết luận mới được đẩy vào GT.
  \item \textbf{Equation Solving (RC5)}: Giải các phương trình toán học và hệ phương trình để tìm các ẩn số mới.
\end{enumerate}

\subsection{Giải phương trình 1D (Brent's Method - RC5)}


KBMS sử dụng thuật toán Brent để tìm nghiệm của phương trình $f(x)=0$ trong một khoảng cô lập.


\subsubsection{Độ hội tụ (Convergence)}

Thuật toán kết hợp giữa Bisection (đảm bảo hội tụ) và Secant/Inverse Quadratic Interpolation (đảm bảo tốc độ).

  \item \textbf{Ưu điểm}: Không yêu cầu tính đạo hàm $f'(x)$.
\end{itemize}




\subsection{Giải hệ phương trình 2D (Newton-Raphson - RC5)}


Được kích hoạt khi phát hiện 2 phương trình có chung 2 biến chưa biết (Dòng code 341-374).


\subsubsection{Ma trận Jacobian \& Công thức cập nhật}

Hệ thống tính toán ma trận Jacobian $\mathbf{J}$ bằng phương pháp sai phân hữu hạn (Dòng code 573):\par
\begin{equation}
 \mathbf{J} = \begin{bmatrix} \frac{\partial f_1}{\partial x} & \frac{\partial f_1}{\partial y} \\ \frac{\partial f_2}{\partial x} & \frac{\partial f_2}{\partial y} \end{bmatrix} 
\end{equation}


Sau đó cập nhật giá trị theo công thức (Dòng code 579-583):\par
\begin{equation}
 \Delta \mathbf{x} = -\mathbf{J}^{-1} \mathbf{F}(\mathbf{x}) 
\end{equation}

KBMS sử dụng một giao thức truyền tin nhị phân (Binary Protocol) tùy chỉnh chạy trên nền TCP để đảm bảo tốc độ truyền tải tri thức nhanh nhất và khả năng xử lý đồng thời (concurrency) tốt.


\subsection{Cấu trúc Gói tin (Packet Structure)}


Mỗi thông điệp (Message) gửi đi giữa Client và Server được đóng gói theo định dạng sau:


\begin{table}[H]
  \centering
  \begin{tabularx}{\textwidth}{|l|X|X|}
  \hline
Trường & Kích thước & Mô tả \\ \hline
\textbf{Total Length} & 4 bytes (Int32) & Tổng độ dài các trường tính từ \textit{SessLen} trở đi (bỏ qua MessageType). \\ \hline
\textbf{Message Type} & 1 byte & Loại thông điệp (Query, Result, Error, ...). \\ \hline
\textbf{SessLen} & 2 bytes (UInt16) & Độ dài của chuỗi Session ID. \\ \hline
\textbf{Session ID} & N bytes (UTF-8) & Định danh phiên làm việc của người dùng. \\ \hline
\textbf{ReqLen} & 2 bytes (UInt16) & Độ dài của chuỗi Request ID. \\ \hline
\textbf{Request ID} & M bytes (UTF-8) & Định danh duy nhất cho mỗi yêu cầu truy vấn. \\ \hline
\textbf{Payload} & P bytes (UTF-8) & Nội dung thực tế (ví dụ: câu lệnh KBQL hoặc kết quả JSON). \\ \hline

\subsection{Các loại Thông điệp (Message Types)}


Máy chủ được trang bị \textbf{12 Giao thức Chuẩn} bao quát mọi tình huống Truy vấn, Phân mảnh dữ liệu, Bảo mật và Telemetry Hệ thống (Thiết kế gốc từ `MessageType.cs`):


\begin{enumerate}
  \item \textbf{Bảo mật \& Xác thực (Security Auth):}
\begin{itemize}
  \item `LOGIN (0x01)`: Xin cấp quyền phiên làm việc bằng thông tin (User/Pass).
  \item `LOGOUT (0x05)`: Đóng kết nối định danh và trả phiên tài nguyên hệ thống.
\end{itemize}
\end{enumerate}



\begin{enumerate}
  \item \textbf{Truy vấn (Query Routing):}
\begin{itemize}
  \item `QUERY (0x02)`: Chở luồng văn bản gốc (Chuỗi lệnh KBQL).
  \item `RESULT (0x03)`: Gói kết quả suy diễn ở dạng tổng quát (JSON String) .
  \item `ERROR (0x04)`: Payload chứa tọa độ `Line`, `Column` và `Stacktrace` dùng để tô đỏ biên dịch viên của Designer.
\end{itemize}
\end{enumerate}



\begin{enumerate}
  \item \textbf{Kỹ thuật Trả Lô Dữ Liệu Lớn (Tabular Data Streaming - TDS)}
    KBMS xử lý Hàng nghìn Fact mà không làm treo RAM Client bằng luồng C\# phân đoạn kinh điển:
\begin{itemize}
  \item `METADATA (0x06)`: Truyền trước sơ đồ lưới dữ liệu (Tên Field, Bề rộng UI Column).
  \item `ROW (0x07)`: Trình bày Payload theo chuỗi vòng lặp "Lô" (Ví dụ: Batch 100 Rows/Lần gói) gửi liên tục qua luồng Stream TCP.
  \item `FETCH\_DONE (0x08)`: Cờ khóa hạ màn chặn chuỗi vòng lặp đọc. Tổng hợp độ trễ thực thi CSDL (Execution Time - ms).
\end{itemize}
\end{enumerate}



\begin{enumerate}
  \item \textbf{Hệ Quản trị Trực tiếp (Telemetry System)}
    Dành riêng cho các Client mang quyền hệ thống cấp cao (ROOT Role):
\begin{itemize}
  \item `STATS (0x0A / 10)`: Theo dõi Metrics máy chủ thời gian thực.
  \item `LOGS\_STREAM (0x0B / 11)`: Đăng ký lắng nghe Live Log Socket đang chảy của Hệ thống.
  \item `SESSIONS (0x0C / 12)`: Trích xuất danh bạ quản lý Users Online, kiểm soát giới hạn Concurrent Connections.
  \item `MANAGEMENT\_CMD (0x0D / 13)`: Đỉnh cao phân quyền - Dùng để chạy lệnh thao tác cấp hệ điều hành (Chỉnh sửa Setting, Kill Session, Cấp Role).
\end{itemize}
\end{enumerate}




\subsection{Thuật toán Xử lý Luồng tin (Streaming Algorithm)}


Để tránh hiện tượng treo ứng dụng khi truyền tải lượng tri thức lớn, KBMS áp dụng cơ chế \textbf{Asynchronous Packet Processing}:


\begin{enumerate}
  \item \textbf{Header Parsing:} Đọc 4 byte đầu tiên để xác định kích thước gói tin cần nhận.
  \item \textbf{Buffer Allocation:} Cấp phát vùng nhớ vừa đủ cho gói tin đó.
  \item \textbf{Exact Reading:} Sử dụng hàm `ReadExact` để đảm bảo nhận đủ dữ liệu trước khi giải mã (Decode).
  \item \textbf{Payload Dispatching:} Chuyển nội dung UTF-8 vào bộ phân tích cú pháp (Parser) hoặc hiển thị lên UI.
\end{enumerate}




\subsection{Bảo mật và Kết nối}


\begin{itemize}
  \item \textbf{TCP Keep-Alive:} Duy trì kết nối ổn định giữa KBMS Studio và Server.
  \item \textbf{Session Management:} Cho phép người dùng thực hiện nhiều truy vấn đồng thời trên cùng một kết nối mà không bị lẫn lộn dữ liệu (nhờ vào Request ID).
  \item \textbf{Error Propagation:} Khi Server gặp lỗi suy diễn, nó sẽ đóng gói toàn bộ Stack Trace và vị trí lỗi vào một `Error Message` đặc biệt để Client hiển thị nháy đỏ trên trình soạn thảo bài toán.
\end{itemize}




\subsection{Kiến trúc Xử lý Chịu tải (Concurrency Threading)}


Để duy trì hiệu năng cao phục vụ cho hàng ngàn điểm kết nối TCP, KBMS Server loại bỏ hoàn toàn mô hình 1-Thread-Per-Client truyền thống, thay vào đó áp dụng cấu trúc xử lý Đa luồng Bất đồng bộ (Async/Await) đỉnh cao của .NET Core:

\begin{itemize}
  \item \textbf{Asynchronous I/O Socket:} Cổng nhận (`AcceptTcpClientAsync`) sẽ quăng toàn bộ tiến trình phân tích Packet vào ThreadPool để tránh tắt nghẽn luồng lắng nghe chính (`\_ = HandleClientAsync(client)`).
  \item \textbf{Tránh Va Chạm Thread Mạng:} Thuật toán gói dữ liệu sử dụng cờ báo `SemaphoreSlim` để khóa luồng ghi byte, tránh trường hợp Response từ hai yêu cầu bất đồng bộ (Async) va chạm và ghi chồng byte lên nhau trên cùng Network Stream.
  \item \textbf{Thread-Safe Sessions:} Việc thu thập và phát tán Request / Log liên quan tới client được lưu trữ tập trung qua lớp `ConnectionManager` bằng bộ từ điển an toàn luồng `ConcurrentDictionary`.
\end{itemize}



Mỗi khi người dùng gửi một câu lệnh KBQL từ Client, Server sẽ thực hiện một quy trình xử lý đa tầng để hiểu ý định (Intent) và thực thi đúng các phép toán/suy diễn.

\subsection{Phân tích Từ vựng (Lexer)}


\textbf{Ý tưởng:} Biến chuỗi văn bản (String) thô thành một danh sách các tín hiệu có nghĩa (Tokens).

\begin{itemize}
  \item \textbf{Thuật toán Tokenizer:} Quét chuỗi ký tự, nhận diện các từ khóa (`SELECT`, `CREATE`), định danh (tên Concept), toán tử và giá trị số/chuỗi.
  \item \textbf{Báo lỗi:} Nếu gặp ký tự lạ, Tokenizer ngay lập tức trả về lỗi kèm vị trí chính xác (Dòng, Cột).
\end{itemize}




\subsection{Phân tích Cú pháp (Parser \& AST)}


\textbf{Ý tưởng:} Xây dựng cây cú pháp trừu tượng (Abstract Syntax Tree - AST) từ danh sách Tokens.

\begin{itemize}
  \item \textbf{Kỹ thuật:} Sử dụng trình phân tích cú pháp \textbf{Recursive Descent} (Phân tích đi xuống đệ quy) để khớp các Tokens với bộ ngữ pháp của KBQL.
  \item \textbf{Cấu trúc AST:} Mỗi câu lệnh được biểu diễn thành một cây các đối tượng. Ví dụ, lệnh `SELECT` sẽ có các nhánh con là `Fields`, `Source`, `WhereCondition`, và `Limit`.
\end{itemize}




\subsection{Thực thi và Cấp phát (Executor \& Router)}


Sau khi có AST từ `KBMS.Parser`, gói tin bắt đầu chuyển giao cho Logic Cấu trúc lõi:

\begin{enumerate}
  \item \textbf{Chồng Khớp (Binding) \& Routing:} Từng lệnh tách lẻ trong chuỗi văn bản được cắt ra và chuyển qua Router `\_knowledgeManager.Execute(ast, ...)` để thực thi tuần tự.
  \item \textbf{Catalog Check:} Concept được truy vấn có tồn tại không?
  \item \textbf{Schema \& Type Check:} Các biến trong hàm `SELECT` hoặc `CALC()` có hợp lệ về kiểu dữ liệu hay không? Mọi sai phạm đều ngay lập tức bị tóm gọn ở lớp `KBMS.Parser.ParserException`, đính kèm vào đó (Line, Column) và đẩy ngược ra Network Client dưới dạng JSON (`Type: ERROR`) nhằm nhắc người dùng sửa sai.
\end{enumerate}




\subsection{Lập kế hoạch Thực thi (Query Execution Plan)}


Hệ thống quyết định cách thức thực hiện truy vấn tối ưu nhất:

\begin{itemize}
  \item \textbf{Access Pattern:} Sử dụng Index (B+ Tree) hay quét tuần tự (Full Scan)?
  \item \textbf{Join Strategy:} Áp dụng phương pháp JOIN nào để kết hợp các Khái niệm?
  \item \textbf{Reasoning Injection:} Nếu truy vấn yêu cầu suy diễn (`INFER` hoặc truy cập các biến suy diễn), hệ thống sẽ tiêm (Inject) thêm các bước gọi vào `ReasoningEngine` trước khi trả kết quả cuối cùng.
\end{itemize}




\subsection{Ví dụ luồng xử lý thực tế}


Câu lệnh: `SELECT name FROM Student WHERE age > 18;`


\begin{enumerate}
  \item \textbf{Lexer:} `[SELECT, name, FROM, Student, WHERE, age, >, 18, ;]`
  \item \textbf{Parser:} Tạo đối tượng `SelectStatement` với `Table=Student`, `Filter=(age > 18)`.
  \item \textbf{Binder:} Xác nhận `Student` có trong KB, `name` và `age` là hợp lệ.
  \item \textbf{Executor:}
\begin{itemize}
  \item Gọi `StorageEngine` để lấy dữ liệu từ B+ Tree của `Student`.
  \item Sử dụng `ReasoningEngine` để lọc các bản ghi có `age > 18`.
  \item Chỉ trả về trường `name` cho phía Network Layer.
\end{itemize}
\end{enumerate}




\subsection{Kế hoạch Triển khai Mã Nguồn Backend (Implementation Strategy)}


Gắn liền với kiến trúc C\# của hệ thống, luồng thiết kế cho tầng mạng và Bộ phân tích từ vựng (Network \& Parser) được thực thi theo 4 Giai đoạn định hình thực tế:


\subsubsection{Giai đoạn 1: Quản trị Kết nối (TCP Listener \& Async Handle)}

Trực tiếp thiết lập máy chủ `KbmsServer` trên kiến trúc Async/Await I/O. Quăng nhiệm vụ hứng Client đang rỗi vào vùng không chặn (Fire-and-forget qua `\_ = HandleClientAsync()`) nhằm khai phóng ThreadPool. Thiết lập hệ thống quét tệp chết (Cleanup Expired Sessions) nhằm đóng ngắt `Socket` lậu và hoàn trả RAM về máy chủ.


\subsubsection{Giai đoạn 2: Mã hóa Giao thức Nhị phân (Packet Binary Protocol)}

Tiến hành lập trình bộ vỏ `KBMNetwork` để viết định dạng truyền tải tuân thủ tài liệu cấu trúc gói `Protocol.cs`. Viết các thao tác C\# Core cấp thấp `Span<byte>`, `BinaryPrimitives` xử lý `BigEndian` tạc khuôn truyền tin (Đọc đủ Byte Session, RequestID, và Payload).


\subsubsection{Giai đoạn 3: Phân tích Cú Pháp Đệ Quy (Lexer \& AST Parser)}

Sử dụng tư duy trình biên dịch tạo module `KBMS.Parser`. Lập trình `Token` và quét nhận dạng Regex cắt chuỗi chữ vô nghĩa thành mảng ký pháp Logic. Xây dựng một cỗ máy phân tích đi xuống (Recursive Descent Parser) quét nối Tokens thành Cây Cú Pháp `AST` (Abstract Syntax Tree), giúp CSDL hiểu ý đồ của con người.


\subsubsection{Giai đoạn 4: Bảo mật \& Phiên Giao dịch (Session \& Security)}

Quản trị hàng ngàn User kết nối TCP đồng thời thông qua tính năng cấu trúc `ConcurrentDictionary` của lớp `ConnectionManager.cs`. Triển khai tính năng User Auth Protocol (LOGIN Message), xác minh Password mã hóa trên hệ từ điển `UserCatalog`, rà soát quyền sở hữu và hạn chế tấn công. Bắt lỗi AST trả về cấu trúc lỗi (kéo theo Line/Col) trên gói JSON Network gửi tới UI của Designer.

\section{Điều phối Mạng \& Luồng Hoạt động}


Máy chủ KBMS (`KbmsServer.cs`) được thiết kế trên mô hình xử lý hiệu năng cao, cho phép tiếp nhận hàng ngàn kết nối đồng thời nhờ vào kiến trúc \textbf{Task-based Asynchronous Pattern (TAP)} của .NET.


\subsection{Mô hình Xử lý Luồng (Async Threading Model)}


Khác với các hệ thống cũ sử dụng `1-Thread-Per-Client`, KBMS sử dụng kỹ thuật bất đồng bộ thực thụ để giải phóng bộ nhớ và tăng khả năng chịu tải.


\begin{figure}[H]
  \centering
  \includegraphics[width=0.7\textwidth]{assets/async_threading_model.png}
  \caption{Mô hình xử lý bất đồng bộ (Asynchronous) sử dụng ThreadPool của .NET.}\label{fig:10_1}
\end{figure}


\begin{enumerate}
  \item \textbf{Chấp nhận kết nối (Acceptance Loop)}:
    Vòng lặp `while` chính sử dụng `await \_listener.AcceptTcpClientAsync(...)`. Khi có client kết nối, hệ thống không đợi mà ngay lập tức spawn một tác vụ con (`\_ = HandleClientAsync(client)`) rồi quay lại lắng nghe client tiếp theo.
  \item \textbf{Xử lý Gói tin (Fire-and-Forget)}:
    Nhiệm vụ xử lý client được đẩy hoàn toàn vào ThreadPool của hệ điều hành. Điều này giúp Main Thread của Server không bao giờ bị nghẽn (Non-blocking).
  \item \textbf{Hàng đợi Ghi (Semaphore Lock)}:
    Do Server có thể gửi nhiều thông điệp bất đồng bộ (ví dụ: đang gửi kết quả Query thì Server muốn gửi thêm Live Log), lớp `Protocol` sử dụng `SemaphoreSlim` để khóa luồng ghi nhị phân, đảm bảo các byte không bị trộn lẫn trên đường truyền.
\end{enumerate}




\subsection{Quản lý Phiên (Session Management)}


Kéo dài sự tồn tại của kết nối và định danh người dùng thông qua lớp `ConnectionManager.cs`.


\begin{itemize}
  \item \textbf{Lưu trữ an toàn luồng}: Toàn bộ các phiên làm việc được lưu trong một `ConcurrentDictionary`. Đây là cấu trúc dữ liệu tối ưu cho phép hàng trăm Thread con cùng lúc đọc/ghi thông tin Session mà không gây ra lỗi tranh chấp bộ nhớ (Race Conditions).
  \item \textbf{Dọn dẹp tài nguyên (Cleanup Task)}:
    Máy chủ khởi chạy một tiến trình ngầm `StartSessionCleanupTask` định kỳ mỗi \textbf{30 giây}. Nó quét qua danh sách `LastActivityAt` và tự động đóng các `Socket` của client đã quá lâu không tương tác (`DefaultTimeoutSeconds`), giúp rảnh RAM cho những kết nối mới.
\end{itemize}




\subsection{Vòng đời của một Giao dịch (Lifecycle)}


Máy chủ vận hành theo một vòng lặp sự kiện (Event Loop) cho mỗi client:


\begin{enumerate}
  \item \textbf{Receive}: Đọc Header (4 byte đầu) để biết kích thước gói.
  \item \textbf{Dispatch}: Chuyển giao gói tin cho `ProcessMessageAsync`.
  \item \textbf{Execute}: Nếu là truy vấn, Server gọi `KnowledgeManager` để xử lý.
  \item \textbf{Respond}: Đóng gói kết quả và gửi ngược lại client.
\end{enumerate}




\section{Kiến trúc Bộ não Điều phối (Knowledge Engine)}


`KnowledgeManager.cs` là thành phần trung tâm, đóng vai trò "Bộ não" của toàn bộ hệ thống KBMS. Nó kết nối giữa Tầng Ngôn ngữ (Parser/AST) và Tầng Lưu trữ/Suy diễn (Storage/Reasoning).


\subsection{Cơ chế Bộ hạ lệnh (The Dispatcher)}


Sau khi `KBMS.Parser` bẻ gãy câu lệnh thành một cây cú pháp (AST), `KnowledgeManager` sẽ thực hiện đệ quy qua danh sách các nút này và gọi hàm `Execute(ast, user, currentKb)`.


\begin{verbatim}
// Luồng thực thi lõi trong KnowledgeManager.cs
public object Execute(AstNode ast, User user, string? currentKb) {
    var kbName = DetermineKbName(ast) ?? currentKb;
    var action = DetermineAction(ast);
    
    if (!CheckPrivilege(user, action, kbName)) 
        return ErrorResponse.PermissionErrorResponse(action, kbName);
        
    return ExecuteQuery(ast, kbName);
}
\end{verbatim}


\subsubsection{Chức năng chính:}

\begin{itemize}
  \item \textbf{Routing (Định tuyến)}: Xác định lệnh này thuộc về Knowledge Base (KB) nào dựa trên từ khóa `USE` hoặc tham số câu lệnh.
  \item \textbf{Security Guard (Bảo vệ)}: Sử dụng `CheckPrivilege` để lọc quyền của `User` hiện tại (ROOT, ADMIN, WRITE, SELECT). Mọi hành động xâm phạm đều bị chặn ngay tại đây trước khi chạm vào dữ liệu.
  \item \textbf{Execution Pipeline}: Hàm `ExecuteQuery` chứa một bộ chuyển mạch (Switch-case) khổng lồ điều hướng hơn \textbf{50 loại nút AST} khác nhau (từ `CREATE\_CONCEPT` đến `SOLVE`).
\end{itemize}




\subsection{Các Công nghệ Liên kết Tri thức (Knowledge Binding)}


Hệ thống điều phối của KBMS V3 mang tính đột phá nhờ các cơ chế sau:


\subsubsection{Auto-expand Variables (Tự nở rộng biến)}

Khi bạn khai báo một Concept chứa thuộc tính là một Concept khác (Ví dụ: `p: Point`), `KnowledgeManager` sẽ tự động "nở" biến này thành các thuộc tính con (`p.x`, `p.y`).

\begin{itemize}
  \item \textbf{Lợi ích}: Giúp bộ máy suy diễn (`ReasoningEngine`) có thể truy cập trực tiếp vào các biến cơ sở mà không cần phải parse lại cấu trúc object phức tạp (Flattening process).

\subsection{Quản lý Giao dịch (Transaction Control)}


Khác với các hệ thống đơn giản, bộ điều phối của bạn hỗ trợ \textbf{Transaction Control Language (TCL)}:

  \item \textbf{ROLLBACK}: Xóa sạch bộ đệm, khôi phục trạng thái tri thức về điểm an toàn trước đó.
\end{itemize}

\subsection{Hệ thống Nhật ký Luồng (Live Log Streaming)}


Đây là tính năng độc đáo nhất của KBMS V3, cho phép "lắng nghe" mọi biến động của máy chủ ngay lập tức.

  \item \textbf{Write to System KB}: Ghi trực tiếp một Fact mới vào Concept `Log` của Knowledge Base `system`.

\begin{itemize}
  \item Bất kỳ dòng log mới nào vừa phát sinh sẽ được "bắn" ngay lập tức vào Socket của client đó dưới dạng gói tin chuẩn.
  \item Điều này cho phép xây dựng các bảng điều khiển (Dashboard) giám sát tri thức mà không cần phải làm mới (Refresh) trang web hay ứng dụng.
\end{itemize}

\end{itemize}

\subsection{Cơ chế Quét (Scanning Mechanism)}


Lexer vận hành như một máy trạng thái (State Machine) với hai con trỏ: `\_start` (điểm bắt đầu của một từ) và `\_current` (vị trí đang quét hiện tại).

\textit{Hình: Quy trình quét chuỗi và định danh Token (dàn ngang)}


\subsubsection{Quy trình xử lý (`ScanToken`):}

\begin{enumerate}
  \item \textbf{Bỏ qua nhiễu}: Loại bỏ khoảng trắng (` `), tab (`\textbackslash{}t`), xuống dòng (`\textbackslash{}n`). Khi gặp xuống dòng, Lexer tự động tăng chỉ số `\_line` và reset `\_column` để phục vụ báo lỗi.
  \item \textbf{Ký tự đơn}: Định danh ngay lập tức các Token như `(`, `)`, `,`, `;`, `+`, `-`, `*`, `/`.
  \item \textbf{Toán tử kép}: Sử dụng hàm `Match()` để kiểm tra ký tự tiếp theo (Ví dụ: `>` và `>=`).
  \item \textbf{Chú thích (Comments)}: Khi gặp `--`, Lexer sẽ bỏ qua toàn bộ phần còn lại của dòng.
\end{enumerate}




\subsection{Xử lý các Hằng số (Literals)}


`Lexer.cs` chứa các phương thức chuyên biệt để bóc tách dữ liệu thực tế:


\subsubsection{Hằng số Số (`NumberLiteral`)}

Hỗ trợ đầy đủ các định dạng số học:

\begin{itemize}
  \item \textbf{Số nguyên}: `123`.
  \item \textbf{Số thực}: `123.45`.
  \item \textbf{Số âm}: Nhận diện dấu `-` đứng trước chữ số.
  \item \textbf{Số mũ (Scientific Notation)}: Hỗ trợ `1.23e-10`.
  \item \textbf{Xử lý nội bộ}: Sử dụng `double.TryParse` với `CultureInfo.InvariantCulture` để đảm bảo tính nhất quán trên mọi hệ điều hành.
\end{itemize}



\subsubsection{Hằng số Chuỗi (`StringLiteral`)}

Hỗ trợ cả nháy đơn `'` và nháy kép `"`:

\begin{itemize}
  \item \textbf{Ký tự thoát (Escape)}: Cho phép dùng nháy kép lặp lại (`""`) để biểu diễn một dấu nháy đơn trong chuỗi.
  \item \textbf{Đa dòng}: Chuỗi có thể kéo dài qua nhiều dòng, Lexer sẽ tự nạp các ký tự xuống dòng vào giá trị literal.
\end{itemize}

\section{Tầng Parser: Cơ chế Đệ quy Đi xuống (Recursive Descent)}


Lớp `Parser.cs` là "ngã tư" điều hướng mọi câu lệnh. Với hơn 3400 dòng mã nguồn, nó thực hiện một quy trình phân tích cú pháp hiệu năng cao dựa trên thuật toán \textbf{Recursive Descent (Đệ quy đi xuống)}.


\subsection{Cơ chế Điều hướng (Sentence Dispatching)}


Mỗi khi máy chủ nhận được một chuỗi Tokens từ Lexer, nó khởi tạo `Parser.cs` và gọi hàm `ParseAll()`.


\begin{verbatim}
// Luồng xử lý đa câu lệnh trong Parser.cs
public List<AstNode> ParseAll() {
    var statements = new List<AstNode>();
    while (!IsAtEnd()) {
        var stmt = ParseStatement(); // "Ngã tư" điều hướng
        statements.Add(stmt);
        if (Check(TokenType.SEMICOLON)) Advance(); // Tiêu thụ dấu ';'
    }
}
\end{verbatim}


\subsubsection{Hàm `ParseStatement()`}

Đây là bộ khung switch-case khổng lồ, quyết định câu lệnh tiếp theo là DDL (Định nghĩa), DML (Truy vấn) hay TCL (Giao dịch). Mỗi khi nhận diện được một từ khóa bắt đầu (như `CREATE`, `SELECT`), nó sẽ ủy quyền cho một hàm con chuyên biệt (ví dụ: `ParseCreate()`, `ParseSelect()`).



\subsection{Kỹ thuật Dự đoán \& Tiêu thụ (Look-ahead)}


KBMS sử dụng ngữ pháp \textbf{LL(k)}, cho phép Parser "nhìn trước" $k$ Token để quyết định nhánh thực thi mà không cần quay lui (Backtracking).

  \caption{Cấu trúc cây phân tích cú pháp (AST) của KBQL}\label{fig:auto_44}
\end{figure}

\textit{Hình: Ví dụ cấu trúc cây AST sau khi phân tích (dàn ngang)}


\begin{itemize}
  \item \textbf{`Peek()`}: Xem Token hiện tại mà không tiêu thụ.
  \item \textbf{`Check(type)`}: Kiểm tra xem Token hiện tại có đúng Type mong muốn hay không.
  \item \textbf{`Consume(type)`}: Nếu Token đúng Type, tiến tới Token tiếp theo. Nếu sai, ngay lập tức ném ra `ParseException` kèm tọa độ `Line/Column`.
  \item \textbf{`Advance()`}: Luôn tiến về phía trước, đảm bảo độ phức tạp thời gian là $O(N)$ (N là số lượng Token).
\end{itemize}




\subsection{Phân tích Khối Lệnh phức tạp (Complex Parsing)}


Một trong những phần xử lý tinh vi nhất trong `Parser.cs` là phân tích lệnh `CREATE CONCEPT`. Thay vì một parser phẳng, nó thực hiện đệ quy lồng nhau:


\begin{enumerate}
  \item \textbf{`ParseCreate`} gọi \textbf{`ParseCreateConcept`}.
  \item \textbf{`ParseCreateConcept`} lặp trong khối `(...)` và kiểm tra từ khóa.
  \item Nếu gặp `VARIABLES`, nó gọi \textbf{`ParseVariableDefinition`} để bóc tách `name: TYPE`.
  \item Nếu gặp `CONSTRAINTS`, nó gọi \textbf{`ParseConstraintList`} (đây là lúc Expression Parser được kích hoạt).
  \item Nếu gặp `RULES`, nó gọi \textbf{`ParseConceptRuleList`} (phân tích IF-THEN logic).
\end{enumerate}

\subsection{Phân tích Biểu thức Nhị phân (Binary Expressions)}


Đây là các phép toán có hai vế, được xử lý thông qua `BinaryExpressionNode`.

  \item \textbf{Toán tử Toán học}: `+`, `-`, `*`, `/`, `\textasciicircum{}`, `\%`.
  \item \textbf{Toán tử So sánh}: `=`, `!=`, `<`, `>`, `<=`, `>=`.
  \item \textbf{Toán tử Logic}: `AND`, `OR`.
\end{itemize}



\subsubsection{Thứ tự ưu tiên (Precedence)}

Parser của KBMS thực hiện phân tích theo mức độ ưu tiên chuẩn quốc tế để đảm bảo kết quả chính xác:

\begin{enumerate}
  \item \textbf{Cấp 1}: Dấu ngoặc `(...)`.
  \item \textbf{Cấp 2}: Hàm gọi và Truy cập thuộc tính (`.`).
  \item \textbf{Cấp 3}: Lũy thừa `\textasciicircum{}`.
  \item \textbf{Cấp 4}: Nhân/Chia `*`, `/`.
  \item \textbf{Cấp 5}: Cộng/Trừ `+`, `-`.
  \item \textbf{Cấp 6}: So sánh `=`, `!=`, `<`, `>`.
  \item \textbf{Cấp 7}: Logic `NOT`, `AND`, `OR`.
\end{enumerate}




\subsection{Các Đơn vị Biểu thức Cốt lõi (Core Expression Units)}


Trong cây AST, biểu thức được phân rã thành các nút lá và nút nhánh:


\subsubsection{`VariableNode` (Biến tri thức)}

Đại diện cho các thuộc tính của Concept (Ví dụ: `a`, `age`). Parser hỗ trợ truy cập thuộc tính lồng nhau thông qua dấu chấm (`p1.x`), giúp xử lý tri thức đa tầng.


\subsubsection{`LiteralNode` (Hằng số)}

Chứa giá trị thực tế như số (`100`), chuỗi (`"Active"`), hoặc Boolean (`true`/`false`). Mỗi nút lưu trữ cả giá trị thô và kiểu dữ liệu (`TokenType`).


\subsubsection{`FunctionCallNode` (Hàm gọi)}

Xử lý các hàm toán học hoặc tri thức tích hợp (v.đ: `Sqrt(x)`, `Abs(y)`). Parser bóc tách tên hàm và một danh sách các tham số (`List<ExpressionNode>`), cho phép các hàm có thể lồng nhau linh hoạt.



\subsection{Xử lý Biểu thức Đơn phân (Unary Expressions)}


Được đại diện bởi `UnaryExpressionNode`, dùng cho các trường hợp chỉ có một vế:

\begin{itemize}
  \item \textbf{Phép phủ định}: `NOT (a > B)`.
  \item \textbf{Số âm}: `-x`.
\end{itemize}

\subsection{Luồng Gửi lệnh (Command Execution Flow)}


Khi người dùng nhấn Enter, CLI thực hiện các bước sau:


\begin{figure}[H]
  \centering
  \includegraphics[width=0.7\textwidth]{assets/cli_input_flow.png}
  \caption{Sơ đồ luồng xử lý đầu vào trong giao diện CLI}\label{fig:auto_52}
\end{figure}


\begin{enumerate}
  \item \textbf{Input Buffering}: Tích lũy các dòng cho đến khi gặp dấu `;`.
  \item \textbf{Request Construction}: Đóng gói lệnh thành các `Message` (Dạng `QUERY` hoặc `LOGIN`).
  \item \textbf{Binary Transport}: Gửi qua `KBMS.Network` (Socket).
  \item \textbf{Wait for Response}: Lắng nghe phản hồi từ máy chủ.
\end{enumerate}




\subsection{Bộ máy Phân tích Phản hồi (`ResponseParser`)}


Phần phức tạp nhất của CLI nằm ở `ResponseParser.cs`. Do kết quả từ Server có thể là một luồng dữ liệu (Streaming ROWS), CLI phải xử lý từng gói tin:


\begin{itemize}
  \item \textbf{METADATA}: Định nghĩa các cột (Name, Type).
  \item \textbf{ROW}: Dữ liệu thực tế cho từng bản ghi.
  \item \textbf{RESULT}: Thông điệp cuối cùng (v.d: "Insert successful").
  \item \textbf{ERROR}: Chứa thông tin lỗi (Message, Line, Column).
\end{itemize}



\subsubsection{Quy trình hiển thị:}

`ResponseParser.cs` thực hiện vẽ bảng "động" theo thuật toán tối ưu:

\begin{enumerate}
  \item \textbf{Header Rendering}: Ngay khi nhận `METADATA`, CLI tính toán độ rộng cột lớn nhất dựa trên tên cột và vẽ khung Header.
  \item \textbf{Multi-line Cell Support}: Nếu dữ liệu trong một ô chứa dấu xuống dòng (`\textbackslash{}n`), CLI sẽ tự động chia ô đó thành nhiều dòng và vẽ đường kẻ phân cách hàng (`|---+---|`) để đảm bảo bảng luôn cân đối.
  \item \textbf{Vertical Rendering}: Đối với các phản hồi thuộc nhóm `Explain\_` hoặc `Describe\_`, CLI chuyển sang chế độ hiển thị theo cặp `Field: Value` trên từng hàng dọc.
\end{enumerate}

CLI hỗ trợ thực thi khối lượng lớn lệnh thông qua tệp tin `.kbql`. Luồng này được xử lý tuần tự để đảm bảo tính nhất quán:

  \caption{Kịch bản thực thi hàng loạt qua lệnh SOURCE trong CLI}\label{fig:auto_53}

\subsection{Đa luồng \& Kết nối (Multi-threading)}


CLI chạy hai luồng chính để duy trì trạng thái:

\begin{enumerate}
  \item \textbf{Main Thread}: Chờ nhận input của người dùng.
  \item \textbf{Heartbeat Thread}: (Tùy chọn) Gửi tín hiệu duy trì kết nối nếu cần thiết.
\end{enumerate}

\begin{itemize}

\begin{enumerate}
  \item \textbf{Gửi lệnh}: Người dùng gõ `DESCRIBE <ConceptName>;`.

\subsection{Kịch bản 2: Tự động hóa Nạp Tri thức (Batch Loading)}


\begin{itemize}
  \item \textbf{Mục tiêu}: Nạp hàng ngàn thực thể tri thức từ tệp tin script có sẵn.
  \item \textbf{Quy trình chi tiết}:
\begin{enumerate}
  \item \textbf{Chuẩn bị}: Tạo tệp `import.kbql` chứa các lệnh `INSERT`.
  \item \textbf{Thực thi}: Gõ `SOURCE import.kbql;` tại CLI.
  \item \textbf{Vòng lặp}: CLI đọc từng block lệnh (phân tách bởi dấu `;`) và gửi tới Server.
  \item \textbf{Xử lý sai sót}: Nếu một lệnh gặp lỗi, CLI sẽ dừng ngay lập tức và chỉ ra vị trí dòng lệnh bị lỗi trong tệp nguồn để quản trị viên sửa chữa.
  \item \textbf{Thống kê}: Sau khi hoàn thành, CLI in ra tổng số bản ghi đã nạp và tổng thời gian thực thi.
\end{enumerate}
\end{itemize}




\subsection{Kịch bản 3: Kiểm tra Kết nối \& An ninh (Security Check)}


\begin{itemize}
  \item \textbf{Mục tiêu}: Đăng nhập và kiểm tra tình trạng kết nối tới máy chủ từ xa.
  \item \textbf{Quy trình}:
\begin{enumerate}
  \item Người dùng gõ `LOGIN <username> <password>`.
  \item CLI gửi gói tin `LOGIN` bảo mật tới Server.
  \item Server xác thực và trả về `LOGIN\_SUCCESS`.
  \item Prompt của CLI chuyển từ `login>` sang `kbms> `, sẵn sàng cho các lệnh truy vấn.
\end{enumerate}
\end{itemize}

\section{Kiến trúc 4 Tầng \& Luồng Xử lý Chi tiết}


KBMS Studio vận hành dựa trên một kiến trúc phân tầng chặt chẽ, từ giao diện người dùng đến các trang dữ liệu nhị phân thấp cấp trên đĩa cứng.


\subsection{Sơ đồ Luồng xử lý 4 Tầng (4-Tier Flow)}


Sơ đồ dưới đây mô tả cách một câu lệnh Tri thức (như `SOLVE`) đi qua hệ thống:

\subsection{Đặc tả các Tầng}


\subsubsection{Tầng 1: Application UI (React/Electron)}

\begin{itemize}
  \item \textbf{Frontend}: React xử lý trạng thái luồng (Redux). Khi người dùng nhấn "Run", lệnh được chuyển xuống Electron Main process.
  \item \textbf{Auth}: Quản lý Session Token và quyền truy cập cục bộ trên giao diện.
\end{itemize}



\subsubsection{Tầng 2: Network \& Protocol (Socket)}

\begin{itemize}
  \item \textbf{Binary Framing}: Chuyển đổi lệnh thành các gói tin nhị phân.
  \item \textbf{Telemetry}: Gửi tín hiệu nhịp tim (Heartbeat) định kỳ để duy trì kết nối Socket ổn định giữa Studio và Server.
\end{itemize}



\subsubsection{Tầng 3: Knowledge Engine (Server)}

\begin{itemize}
  \item \textbf{KnowledgeManager}: Điều phối việc phân tách câu lệnh bằng Lexer/Parser.
  \item \textbf{Reasoning}: Nếu là lệnh `SOLVE`, bộ máy suy diễn sẽ thực hiện Forward Chaining trên các Page dữ liệu được nạp vào RAM.
  \item \textbf{Logs}: Ghi lại nhật ký truy cập (Audit) đồng thời Stream Log trở lại Studio UI.
\end{itemize}



\subsubsection{Tầng 4: Storage Layer (Binary Data)}

\begin{itemize}
  \item \textbf{B+ Tree}: Tìm kiếm bản ghi Object dựa trên Index.
  \item \textbf{LRU Cache}: Quản lý các trang dữ liệu (Pages) trong bộ nhớ đệm để tối ưu hóa tốc độ.
  \item \textbf{WAL (Write-Ahead Logging)}: Đảm bảo tính toàn vẹn tri thức ngay cả khi mất điện đột ngột.
\end{itemize}




\subsection{Luồng Thông báo Thời gian thực (Notification Flow)}


Khác với luồng Request-Response thông thường, hệ thống Notification sử dụng cơ chế \textbf{Server Push}:


\begin{figure}[H]
  \centering
  \includegraphics[width=0.7\textwidth]{assets/4_tier_notification_flow.png}
  \caption{Cơ chế Server Push cho các thông báo hệ thống thời gian thực.}\label{fig:13_2}
\end{figure}


\begin{enumerate}
  \item \textbf{Trigger}: Một sự kiện an ninh hoặc hệ thống xảy ra tại Server.
  \item \textbf{Push}: Server đóng gói `MessageType.NOTIFICATION` và đẩy qua Socket.
  \item \textbf{Dispatch}: Electron Main nhận gói tin và gửi vào Redux Store của React.
  \item \textbf{UI Feedback}: Component `NotificationToasts` hiển thị thông báo popup, đồng thời cập nhật số lượng tin nhắn mới tại `NotificationBell`.
\end{enumerate}




\subsection{Quy trình Đăng nhập (Authentication Flow)}


\begin{enumerate}
  \item \textbf{Handshake}: Studio gửi gói tin `LOGIN` kèm User/Pass được mã hóa Base64.
  \item \textbf{Server Verify}: Server kiểm tra trong hệ thống quản lý User (Tầng 4).
  \item \textbf{Session Record}: Nếu khớp, Server tạo một `SessionContext` trong RAM (Tầng 3) và trả về thông báo thành công.
  \item \textbf{Prompt Update}: Studio cập nhật trạng thái đã đăng nhập và cho phép thao tác với Knowledge Base.
\end{enumerate}
